## 主要作用

**Unity有自带的计时器，继承MonoBehaviour的组件，可以利用**
1. Invoke 延迟执行逻辑
2. InvokeRepeating 间隔一定时间执行逻辑
3. 协同程序 来计时
4. Update中 自己实现计时功能

**但是**
1. 使用Unity自带的计时功能，不能为所有类服务（只能继承了MonoBehaviour才能使用）
2. 利用协同程序和Update制作计时功能，如果需要都写，会产生代码冗余（写一堆类似重复的代码）

**因此**
将计时功能统一的进行管理，让所有系统都能够通过计时器模块进行计时

---

## 基本原理

**想要达到的效果：**
1. 能够延时响应逻辑（比如：10s后执行某一逻辑）
2. 在延时过程中，可以固定时间间隔响应逻辑（比如：10s后执行某一逻辑，但是10s内每隔1s又能响应其它逻辑）

**基本原理：**
 1. 定义一个计时器对象，计时器对象记录延时执行时间、间隔执行时间、执行结束回调委托、间隔执行回调委托等等
2. 计时器模块中统一管理所有计时器对象，每个计时器对象有一个唯一ID，用来区分获取计时器对象
3. 在计时器模块中利用一个协同程序固定间隔时间进行计时器对象的时间检测
4. 计时器模块中提供创建计时器、移除计时器、停止计时器、开始计时器、重置计时器的相关方法

**主要要实现的类：**

**1. 计时器模块管理器 TimerMgr**
  **需要声明的关键成员**
1. 计时器字典容器
2. 待移除计时器列表容器
3. 等等

  **需要实现的关键方法**
1. 开启计时器管理器
2. 关闭计时器管理器
3. 创建单个计时器
4. 移除单个计时器
5. 重置单个计时器
6. 开启单个计时器
7. 停止单个计时器
8. 等等

**2. 计时器对象类 TimerItem**
  **需要声明的关键成员**
1. 唯一ID
2. 延时执行回调委托
3. 间隔执行回到委托
4. 总时间
5. 间隔时间
6. 是否开启
7. 等等

  **需要实现的关键方法**
1. 初始化数据
2. 重置时间
3. 等等

## 基本实现

```csharp
/// <summary>
/// 计时器数据结构 时间统一ms
/// </summary>
public class TimerItem : IPoolObject
{
    /// <summary>
    /// 计时器唯一ID
    /// </summary>
    public int keyID;
    /// <summary>
    /// 延时执行的回调委托
    /// </summary>
    public UnityAction overCallBack;
    /// <summary>
    /// 间隔执行的回调委托
    /// </summary>
    public UnityAction callBack;
    /// <summary>
    /// 总时间
    /// </summary>
    public int allTime;
    /// <summary>
    /// 记录一开始计时时的总时间
    /// </summary>
    private int maxAllTime;
    /// <summary>
    /// 间隔执行回调的时间
    /// </summary>
    public int intervalTime;
    /// <summary>
    /// 记录一开始的间隔时间
    /// </summary>
    public int maxIntervalTime;
    /// <summary>
    /// 是否正在进行计时
    /// </summary>
    public bool isRunning;
    /// <summary>
    /// 初始化计时器的方法
    /// </summary>
    /// <param name="keyID">唯一ID</param>
    /// <param name="allTime">总的时间</param>
    /// <param name="overCallBack">总时间计时结束后的回调</param>
    /// <param name="intervalTime">间隔执行的时间</param>
    /// <param name="callBack">间隔执行时间结束后的回调</param>
    public void InitInfo(int keyID, int allTime, UnityAction overCallBack, int intervalTime = 0, UnityAction callBack = null)
    {
        this.keyID = keyID;
        this.allTime = maxAllTime = allTime;
        this.overCallBack = overCallBack;
        this.intervalTime = maxIntervalTime = intervalTime;
        this.callBack = callBack;
        isRunning = true;
    }
    /// <summary>
    /// 重置计时器
    /// </summary>
    public void ResetTimer()
    {
        allTime = maxAllTime;
        intervalTime = maxIntervalTime;
        isRunning = true;
    }
    public void ResetInfo()
    {
        overCallBack = null;
        callBack = null;
    }
}
/// <summary>
/// 计时器管理器
/// </summary>
public class TimerMgr : BaseManager<TimerMgr>
{
    private Dictionary<int, TimerItem> timerDic = new Dictionary<int, TimerItem>();
    private List<TimerItem> delList = new List<TimerItem>();//待移除列表
    private Coroutine timer;//计时器的协程
    private int TIMER_KEY = 0;//记录当前将要创建的唯一ID
    /// <summary>
    /// 计时器管理器中的唯一计时用的协同程序的 间隔时间
    /// </summary>
    public float intervalTime = 0.1f;
    private TimerMgr() 
    {
        Start();//开启计时器
    }
    
    /// <summary>
    /// 开启计时器的方法
    /// </summary>
    public void Start()
    {
        timer = MonoMgr.Instance.StartCoroutine(StartTiming());
    }
    /// <summary>
    /// 关闭计时器的方法
    /// </summary>
    public void Stop()
    {
        MonoMgr.Instance.StopCoroutine(timer);
    }
    /// <summary>
    /// 计时的协同程序方法
    /// </summary>
    /// <returns></returns>
    IEnumerator StartTiming()
    {
        while (true)
        {
            yield return new WaitForSeconds(intervalTime);//100ms计时一次 可手动调
            foreach (TimerItem item in timerDic.Values) //遍历所有的计时器 更新数据
            {
                if (!item.isRunning) //不在运行状态 跳过
                    continue;
                if (item.callBack != null) //判断是否有间隔时间执行回调的需求
                {
                    item.intervalTime -= (int)(intervalTime * 1000);//减去100ms 与计时器刷新频率同步
                    if (item.intervalTime <= 0)
                    {
                        item.callBack();//执行一次回调
                        item.intervalTime = item.maxIntervalTime;//重置间隔时间
                    }
                }
                item.allTime -= (int)(intervalTime * 1000);
                if (item.allTime <= 0)
                {
                    item.overCallBack();
                    delList.Add(item);//foreach直接移除会报错 先加入待移除队列
                }
            }
            for (int i = 0; i < delList.Count; i++)
            {
                timerDic.Remove(delList[i].keyID);//从字典中移除
                PoolMgr.Instance.PushObj(delList[i]);//放入缓存池
            }
            delList.Clear();//清空列表
        }
    }
    /// <summary>
    /// 创建一个计时器的方法
    /// </summary>
    /// <param name="allTime">总时间</param>
    /// <param name="overCallBack">总时间结束回调</param>
    /// <param name="intervalTime">间隔时间</param>
    /// <param name="callBack">间隔时间结束回调</param>
    /// <returns>返回唯一ID</returns>
    public int CreateTimer(int allTime, UnityAction overCallBack, int intervalTime = 0, UnityAction callBack = null)
    {
        int keyID = ++TIMER_KEY;//构建唯一ID
        TimerItem timerItem = PoolMgr.Instance.GetObj<TimerItem>();//从缓存池中取出计时器
        timerItem.InitInfo(keyID, allTime, overCallBack, intervalTime, callBack); //初始化计时器
        timerDic.Add(keyID, timerItem);//记录到字典中
        return keyID;
    }
    /// <summary>
    /// 移除计时器
    /// </summary>
    /// <param name="keyID">计时器ID</param>
    public void RemoveTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
        {
            PoolMgr.Instance.PushObj(timerDic[keyID]);//放入缓存池
            timerDic.Remove(keyID); //从字典中移除
        }
    }
    /// <summary>
    /// 重置单个计时器
    /// </summary>
    /// <param name="keyID">计时器ID</param>
    public void ResetTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
        {
            timerDic[keyID].ResetTimer();
        }
    }
    /// <summary>
    /// 继续单个计时器
    /// </summary>
    /// <param name="keyID"></param>
    public void StartTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
        {
            timerDic[keyID].isRunning = true;
        }
    }
    /// <summary>
    /// 暂停单个计时器
    /// </summary>
    /// <param name="keyID"></param>
    public void PauseTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
        {
            timerDic[keyID].isRunning = false;
        }
    }
}
```

**遗留问题：**
目前的计时器管理器会受到 `Time.timeScale` 的影响

## 优化

Unity也提供了不受`Time.timeScale`影响的 `yield return new WaitForSecondsRealtime(秒)`

利用它制作不受 `timeScale` 影响的计时器

**主要优化思路：**
在计时器模块中，保留之前的受`Time.timeScale`影响的计时器
并添加一种不受其影响的计时器，让开发者可以根据需求选择使用

**主要修改处：**
1. 添加一个字典容器专门记录不受其影响的计时器
2. 多开一个协同程序专门用于处理不受其影响的计时器
3. 修改相关方法

```csharp
/// <summary>
/// 计时器管理器
/// </summary>
public class TimerMgr : BaseManager<TimerMgr>
{
    private Dictionary<int, TimerItem> timerDic = new Dictionary<int, TimerItem>();//存储受timeScale影响的计时器
    private Dictionary<int, TimerItem> realTimerDic = new Dictionary<int, TimerItem>();//存储不受timeScale影响的计时器
    private List<TimerItem> delList = new List<TimerItem>();//待移除列表
    private Coroutine timer;//计时器的协程
    private Coroutine realTimer;//真实时间计时器的协程
    private int TIMER_KEY = 0;//记录当前将要创建的唯一ID
    private WaitForSecondsRealtime waitForSecondsRealtime = new WaitForSecondsRealtime(intervalTime);//节约内存
    private WaitForSeconds waitForSeconds = new WaitForSeconds(intervalTime);//节约内存
    /// <summary>
    /// 计时器管理器中的唯一计时用的协同程序的 间隔时间
    /// </summary>
    public const float intervalTime = 0.1f;
    private TimerMgr() 
    {
        Start();//开启计时器
    }
    
    /// <summary>
    /// 开启计时器的方法
    /// </summary>
    public void Start()
    {
        timer = MonoMgr.Instance.StartCoroutine(StartTiming(false, timerDic));
        realTimer = MonoMgr.Instance.StartCoroutine(StartTiming(true, realTimerDic));
    }
    /// <summary>
    /// 关闭计时器的方法
    /// </summary>
    public void Stop()
    {
        MonoMgr.Instance.StopCoroutine(timer);
        MonoMgr.Instance.StopCoroutine(realTimer);
    }
    /// <summary>
    /// 计时的协同程序方法
    /// </summary>
    /// <returns></returns>
    IEnumerator StartTiming(bool isRealTime, Dictionary<int, TimerItem> timerDic)
    {
        while (true)
        {
            if (isRealTime)
                yield return waitForSecondsRealtime;//100ms计时一次 可手动调
            else
                yield return waitForSeconds;//100ms计时一次 可手动调
            foreach (TimerItem item in timerDic.Values) //遍历所有的计时器 更新数据
            {
                if (!item.isRunning) //不在运行状态 跳过
                    continue;
                if (item.callBack != null) //判断是否有间隔时间执行回调的需求
                {
                    item.intervalTime -= (int)(intervalTime * 1000);//减去100ms 与计时器刷新频率同步
                    if (item.intervalTime <= 0)
                    {
                        item.callBack();//执行一次回调
                        item.intervalTime = item.maxIntervalTime;//重置间隔时间
                    }
                }
                item.allTime -= (int)(intervalTime * 1000);
                if (item.allTime <= 0)
                {
                    item.overCallBack();
                    delList.Add(item);//foreach直接移除会报错 先加入待移除队列
                }
            }
            for (int i = 0; i < delList.Count; i++)
            {
                timerDic.Remove(delList[i].keyID);//从字典中移除
                PoolMgr.Instance.PushObj(delList[i]);//放入缓存池
            }
            delList.Clear();//清空列表
        }
    }
    /// <summary>
    /// 创建一个计时器的方法
    /// </summary>
    /// <param name="isRealTime">是否不受timeScale影响</param>
    /// <param name="allTime">总时间</param>
    /// <param name="overCallBack">总时间结束回调</param>
    /// <param name="intervalTime">间隔时间</param>
    /// <param name="callBack">间隔时间结束回调</param>
    /// <returns>返回唯一ID</returns>
    public int CreateTimer(bool isRealTime, int allTime, UnityAction overCallBack, int intervalTime = 0, UnityAction callBack = null)
    {
        int keyID = ++TIMER_KEY;//构建唯一ID
        TimerItem timerItem = PoolMgr.Instance.GetObj<TimerItem>();//从缓存池中取出计时器
        timerItem.InitInfo(keyID, allTime, overCallBack, intervalTime, callBack); //初始化计时器
        if (isRealTime)
            realTimerDic.Add(keyID, timerItem);//记录到字典中 
        else
            timerDic.Add(keyID, timerItem);//记录到字典中
        return keyID;
    }
    /// <summary>
    /// 移除计时器
    /// </summary>
    /// <param name="keyID">计时器ID</param>
    public void RemoveTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
        {
            PoolMgr.Instance.PushObj(timerDic[keyID]);//放入缓存池
            timerDic.Remove(keyID); //从字典中移除
        }
        else if (realTimerDic.ContainsKey(keyID)) //如果字典中有
        {
            PoolMgr.Instance.PushObj(realTimerDic[keyID]);//放入缓存池
            realTimerDic.Remove(keyID); //从字典中移除
        }
    }
    /// <summary>
    /// 重置单个计时器
    /// </summary>
    /// <param name="keyID">计时器ID</param>
    public void ResetTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
            timerDic[keyID].ResetTimer();
        else if (realTimerDic.ContainsKey(keyID)) //如果字典中有
            realTimerDic[keyID].ResetTimer();
    }
    /// <summary>
    /// 继续单个计时器
    /// </summary>
    /// <param name="keyID"></param>
    public void StartTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
            timerDic[keyID].isRunning = true;
        else if (realTimerDic.ContainsKey(keyID)) //如果字典中有
            realTimerDic[keyID].isRunning = true;
    }
    /// <summary>
    /// 暂停单个计时器
    /// </summary>
    /// <param name="keyID"></param>
    public void PauseTimer(int keyID)
    {
        if (timerDic.ContainsKey(keyID)) //如果字典中有
            timerDic[keyID].isRunning = false;
        else if (realTimerDic.ContainsKey(keyID)) //如果字典中有
            realTimerDic[keyID].isRunning = false;
    }
}
```
