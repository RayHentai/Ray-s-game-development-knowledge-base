## 主要作用

制作输入控制模块主要是**降低输入相关代码的耦合性**
以前制作老输入系统相关功能，都是在Update当中进行按键检测，处理对应逻辑

但是当游戏开发中存在角色切换功能时，可能就会存在冗余代码
所以将通过输入控制模块来降低代码耦合性，减少冗余代码

---

## 制作思路

1. 制作InputMgr单例模式管理器
2. 在输入管理器中进行按键检测
3. 利用事件中心分发事件
4. 在希望处理输入逻辑的位置监听事件

---

## 基本实现

监听键盘、鼠标、热键的输入
并分发事件

```csharp
public class InputMgr : BaseManager<InputMgr>
{
    private InputMgr() 
    {
        MonoMgr.Instance.AddUpdateListener(InputUpdate);
    }
    private bool isStart = false;
    public void StartOrCloseInputMgr(bool isStart)
    {
        this.isStart = isStart;
    }
    /// <summary>
    /// Mono公共模块实现的帧更新方法
    /// </summary>
    private void InputUpdate()
    {
	    if (!isStart) //如果没有开启输入模块 无需检测
		    return;
        CheckKeyCode(KeyCode.W);
        CheckKeyCode(KeyCode.A);
        CheckKeyCode(KeyCode.S);
        CheckKeyCode(KeyCode.D);
        CheckKeyCode(KeyCode.H);
        CheckKeyCode(KeyCode.J);
        CheckKeyCode(KeyCode.K);
        CheckKeyCode(KeyCode.L);
        //...
        CheckMouse(0);
        CheckMouse(1);
        CheckMouse(2);
        //...
        EventCenter.Instance.EventTrigger(E_EventType.E_Horizontal, Input.GetAxis("Horizontal"));
        EventCenter.Instance.EventTrigger(E_EventType.E_Vertical, Input.GetAxis("Vertical"));
    }
    /// <summary>
    /// 检测键盘输入
    /// </summary>
    /// <param name="key"></param>
    private void CheckKeyCode(KeyCode key) 
    {
        if (Input.GetKeyDown(key)) 
            EventCenter.Instance.EventTrigger(E_EventType.E_KeyboardDown, key);
        if (Input.GetKeyUp(key))
            EventCenter.Instance.EventTrigger(E_EventType.E_KeyboardUp, key);
        if (Input.GetKey(key))
            EventCenter.Instance.EventTrigger(E_EventType.E_Keyboard, key);
    }
    /// <summary>
    /// 检测鼠标输入
    /// </summary>
    /// <param name="key"></param>
    private void CheckMouse(int key)
    {
        if (Input.GetMouseButtonDown(key))
            EventCenter.Instance.EventTrigger(E_EventType.E_MouseDown, key);
        if (Input.GetMouseButtonUp(key))
            EventCenter.Instance.EventTrigger(E_EventType.E_MouseUp, key);
        if (Input.GetMouseButton(key))
            EventCenter.Instance.EventTrigger(E_EventType.E_Mouse, key);
    }
}
```

### 改键功能

**需求分析**
**1.改建功能应该是针对某一个行为的**
- 游戏中的行为是固定的，我们需要改变的是触发该行为的按键
- 比如：角色有技能1、技能2、技能3，我们改变的是触发这些行为的输入

**2.具体键位的触发不应该写死**
- 而是需要根据存储的键位数据进行初始化或进行修改

**3.键位的修改可以是键盘输入，也可以是鼠标输入，输入类型也可以是任意的**
- 比如：触发技能1，可以修改为键盘输入，也可以修改为鼠标输入
- 可以是按下、可以是抬起、也可以是长按，具体是哪种应该根据你的需求变化

**制作思路**
**输入管理器主要做的事情**
- 根据输入的信息，触发对应的事件
- 其中输入信息可变，触发的事件也可变

**1.改建功能应该是针对某一个行为的**
- 触发的事件类型应该是针对行为的，因为行为一般在游戏中是固定的

**2.具体键位的触发不应该写死**
- 输入管理器中应该声明字典容器
- 键：触发事件的类型
- 值：具体的输入信息（键盘还是鼠标、按下还是抬起还是长按、那个键）

**3.键位的修改可以是键盘输入，也可以是鼠标输入，输入类型也可以是任意的**
- 输入管理器应该提供初始化行为对应的键位方法

```csharp
/// <summary>
/// 输入信息
/// </summary>
public class InputInfo
{
    /// <summary>
    /// 键盘还是鼠标
    /// </summary>
    public enum E_KeyOrMouse
    {
        Key,
        Mouse,
    }
    /// <summary>
    /// 输入行为类型
    /// </summary>
    public enum E_InputType
    {
        Down,
        Up,
        Always,
    }
    public E_KeyOrMouse keyOrMouse;//键盘还是鼠标
    public E_InputType inputType;//输入行为
    public KeyCode key;//输入键
    public int mouseID;//鼠标行为id
    /// <summary>
    /// 初始化信息
    /// </summary>
    /// <param name="keyOrMouse">键盘还是鼠标</param>
    /// <param name="inputType">输入行为类型</param>
    /// <param name="key">键</param>
    public InputInfo(E_InputType inputType, KeyCode key, E_KeyOrMouse keyOrMouse = E_KeyOrMouse.Key)
    {
        this.keyOrMouse = keyOrMouse;
        this.inputType = inputType;
        this.key = key;
    }
    /// <summary>
    /// 初始化信息
    /// </summary>
    /// <param name="keyOrMouse">键盘还是鼠标</param>
    /// <param name="inputType">输入行为类型</param>
    /// <param name="mouseID"></param>
    public InputInfo(E_InputType inputType, int mouseID, E_KeyOrMouse keyOrMouse = E_KeyOrMouse.Mouse)
    {
        this.keyOrMouse = keyOrMouse;
        this.inputType = inputType;
        this.mouseID = mouseID;
    }
}

public class InputMgr : BaseManager<InputMgr>
{
    private InputMgr()
    {
        MonoMgr.Instance.AddUpdateListener(InputUpdate);
    }
    //记录输入信息的容器
    public Dictionary<E_EventType, InputInfo> inputDic = new Dictionary<E_EventType, InputInfo>();
    private InputInfo nowInputInfo;
    private bool isStart = false;//输入模块的开关
    /// <summary>
    /// 启动输入模块的方法
    /// </summary>
    /// <param name="isStart"></param>
    public void StartOrCloseInputMgr(bool isStart)
    {
        this.isStart = isStart;
    }
    /// <summary>
    /// 提供给外部改键或初始化的方法（键盘）
    /// </summary>
    /// <param name="eventType">触发事件的类型</param>
    /// <param name="InputType">输入类型</param>
    /// <param name="key">键</param>
    public void ChangeKeyboardInfo(E_EventType eventType, InputInfo.E_InputType InputType, KeyCode key)
    {
        if (!inputDic.ContainsKey(eventType)) //如果没有存储该行为对应的信息
        {
            inputDic.Add(eventType, new InputInfo(InputType, key));
        }
        else //改键
        {
            inputDic[eventType].keyOrMouse = InputInfo.E_KeyOrMouse.Key;//如果上一次是鼠标 改成键盘
            inputDic[eventType].inputType = InputType;//修改输入行为
            inputDic[eventType].key = key;//修改键
        }
    }
    /// <summary>
    /// 提供给外部改键或初始化的方法（鼠标）
    /// </summary>
    /// <param name="eventType">触发事件的类型</param>
    /// <param name="InputType">输入类型</param>
    /// <param name="mouseID">键</param>
    public void ChangeMouseInfo(E_EventType eventType, InputInfo.E_InputType InputType, int mouseID)
    {
        if (!inputDic.ContainsKey(eventType)) //如果没有存储该行为对应的信息
        {
            inputDic.Add(eventType, new InputInfo(InputType, mouseID));
        }
        else //改键
        {
            inputDic[eventType].keyOrMouse = InputInfo.E_KeyOrMouse.Mouse;//如果上一次是键盘 改成鼠标
            inputDic[eventType].inputType = InputType;//修改输入行为
            inputDic[eventType].mouseID = mouseID;//修改键
        }
    }
    /// <summary>
    /// 移除输入信息的方法
    /// </summary>
    /// <param name="eventType">事件类型</param>
    public void RemoveInputInfo(E_EventType eventType)
    {
        if (inputDic.ContainsKey(eventType))
            inputDic.Remove(eventType);
    }
    /// <summary>
    /// Mono公共模块实现的帧更新方法
    /// </summary>
    private void InputUpdate()
    {
        if (!isStart) //如果没有开启输入模块 无需检测
            return;
        foreach (E_EventType eventType in inputDic.Keys) //遍历字典容器 设置事件触发器
        {
            nowInputInfo = inputDic[eventType];
            if (nowInputInfo.keyOrMouse == InputInfo.E_KeyOrMouse.Key) //如果是键盘输入
            {
                switch (nowInputInfo.inputType)
                {
                    case InputInfo.E_InputType.Down:
                        if (Input.GetKeyDown(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                            break;
                    case InputInfo.E_InputType.Up:
                        if (Input.GetKeyUp(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Always:
                        if (Input.GetKey(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                }
            }
            else //如果是鼠标输入
            {
                switch (nowInputInfo.inputType)
                {
                    case InputInfo.E_InputType.Down:
                        if (Input.GetMouseButtonDown(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Up:
                        if (Input.GetMouseButtonUp(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Always:
                        if (Input.GetMouseButton(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                }
            }
        }
        EventCenter.Instance.EventTrigger(E_EventType.InputHorizontal, Input.GetAxis("Horizontal"));
        EventCenter.Instance.EventTrigger(E_EventType.InputVertical, Input.GetAxis("Vertical"));
    }
}
```

**遗留问题：**
在真正制作改建功能时，应该怎么获取到任意键盘或任意鼠标输入来记录对应的输入信息

### 获取输入信息

在InputMgr的更新函数中
获取当前输入内容
用委托返回给外部

**主要思路：**
当存在按下输入时
遍历监听KeyCode枚举中所有键位，检测是哪个键位输入了
监听鼠标左中右键键位，检测是哪个键位输入了

由于开启检测输入时可能伴随着键盘或者鼠标输入
因此利用协同程序延迟一帧再进行检测

```csharp

/// 输入信息
/// </summary>
public class InputInfo
{
    /// <summary>
    /// 键盘还是鼠标
    /// </summary>
    public enum E_KeyOrMouse
    {
        Key,
        Mouse,
    }
    /// <summary>

    /// </summary>

        Down,
        Up,
        Always,
    }

/// <summary>
/// 输入控制模块
/// </summary>
public class InputMgr : BaseManager<InputMgr>
{
    private InputMgr()
    {
        MonoMgr.Instance.AddUpdateListener(InputUpdate);
    }
    //记录输入信息的容器
    public Dictionary<E_EventType, InputInfo> inputDic = new Dictionary<E_EventType, InputInfo>();
    private InputInfo nowInputInfo;
    private bool isStart = false;//输入模块的开关
    private UnityAction<InputInfo> getInputInfoCallBack;//用于改键时返回外部当前输入信息的委托
    private bool isBeginCheckInfo = false;//是否开始检测信息
    /// <summary>
    /// 获取下一帧的输入信息
    /// </summary>
    /// <param name="callback">回调函数</param>
    public void GetInputInfo(UnityAction<InputInfo> callback)
    {
        //记录回调函数
        getInputInfoCallBack = callback;
        //开启协程
        MonoMgr.Instance.StartCoroutine(BeginCheckInput());
    }
    /// <summary>
    /// 等待一帧 开始检测信息的协程方法
    /// </summary>
    /// <returns></returns>
    private IEnumerator BeginCheckInput()
    {
        yield return 0;//等待一帧
        isBeginCheckInfo = true;
    }
    /// <summary>
    /// 启动输入模块的方法
    /// </summary>
    /// <param name="isStart"></param>
    public void StartOrCloseInputMgr(bool isStart)
    {
        this.isStart = isStart;
    }
    /// <summary>
    /// 提供给外部改键或初始化的方法（键盘）
    /// </summary>
    /// <param name="eventType">触发事件的类型</param>
    /// <param name="InputType">输入类型</param>
    /// <param name="key">键</param>
    public void ChangeKeyboardInfo(E_EventType eventType, InputInfo.E_InputType InputType, KeyCode key)
    {
        if (!inputDic.ContainsKey(eventType)) //如果没有存储该行为对应的信息
        {
            inputDic.Add(eventType, new InputInfo(InputType, key));
        }
        else //改键
        {
            inputDic[eventType].keyOrMouse = InputInfo.E_KeyOrMouse.Key;//如果上一次是鼠标 改成键盘
            inputDic[eventType].inputType = InputType;//修改输入行为
            inputDic[eventType].key = key;//修改键
        }
    }
    /// <summary>
    /// 提供给外部改键或初始化的方法（鼠标）
    /// </summary>
    /// <param name="eventType">触发事件的类型</param>
    /// <param name="InputType">输入类型</param>
    /// <param name="mouseID">键</param>
    public void ChangeMouseInfo(E_EventType eventType, InputInfo.E_InputType InputType, int mouseID)
    {
        if (!inputDic.ContainsKey(eventType)) //如果没有存储该行为对应的信息
        {
            inputDic.Add(eventType, new InputInfo(InputType, mouseID));
        }
        else //改键
        {
            inputDic[eventType].keyOrMouse = InputInfo.E_KeyOrMouse.Mouse;//如果上一次是键盘 改成鼠标
            inputDic[eventType].inputType = InputType;//修改输入行为
            inputDic[eventType].mouseID = mouseID;//修改键
        }
    }
    /// <summary>
    /// 移除输入信息的方法
    /// </summary>
    /// <param name="eventType">事件类型</param>
    public void RemoveInputInfo(E_EventType eventType)
    {
        if (inputDic.ContainsKey(eventType))
            inputDic.Remove(eventType);
    }
    /// <summary>
    /// Mono公共模块实现的帧更新方法
    /// </summary>
    private void InputUpdate()
    {
        if (isBeginCheckInfo) //开始检测输入信息
        {
            if (Input.anyKeyDown) //如果有任意键按下
            {
                InputInfo inputInfo = null;
                //遍历所有的键 得到输入的键
                Array keyCodes = Enum.GetValues(typeof(KeyCode)); //这个方法用于得到枚举中的所有值 返回值是Array 
                foreach (KeyCode inputKey in keyCodes)
                {
                    if (Input.GetKeyDown(inputKey))
                    {
                        inputInfo = new InputInfo(InputInfo.E_InputType.Down, inputKey);
                        break;
                    }
                }
                //遍历所有鼠标输入
                for (int i = 0; i < 3; i++)
                {
                    if (Input.GetMouseButtonDown(i))
                    {
                        inputInfo = new InputInfo(InputInfo.E_InputType.Down, i);
                        break;
                    }
                }
                getInputInfoCallBack?.Invoke(inputInfo);
                getInputInfoCallBack = null;
                isBeginCheckInfo = false; //检测信息只执行一次 下次一启动需要重新开启协程
            }
        }
        
        if (!isStart) //如果没有开启输入模块 无需检测
            return;
        foreach (E_EventType eventType in inputDic.Keys) //遍历字典容器 设置事件触发器
        {
            nowInputInfo = inputDic[eventType];
            if (nowInputInfo.keyOrMouse == InputInfo.E_KeyOrMouse.Key) //如果是键盘输入
            {
                switch (nowInputInfo.inputType)
                {
                    case InputInfo.E_InputType.Down:
                        if (Input.GetKeyDown(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                            break;
                    case InputInfo.E_InputType.Up:
                        if (Input.GetKeyUp(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Always:
                        if (Input.GetKey(nowInputInfo.key))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                }
            }
            else //如果是鼠标输入
            {
                switch (nowInputInfo.inputType)
                {
                    case InputInfo.E_InputType.Down:
                        if (Input.GetMouseButtonDown(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Up:
                        if (Input.GetMouseButtonUp(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                    case InputInfo.E_InputType.Always:
                        if (Input.GetMouseButton(nowInputInfo.mouseID))
                            EventCenter.Instance.EventTrigger(eventType);
                        break;
                }
            }
        }
        EventCenter.Instance.EventTrigger(E_EventType.InputHorizontal, Input.GetAxis("Horizontal"));
        EventCenter.Instance.EventTrigger(E_EventType.InputVertical, Input.GetAxis("Vertical"));
    }
}
```

```csharp title:测试
            InputMgr.Instance.GetInputInfo((info) =>
            {
                if (info.keyOrMouse == InputInfo.E_KeyOrMouse.Key) //按下了键盘
                {
                    Debug.Log("按键修改为：" + info.key);
                    InputMgr.Instance.ChangeKeyboardInfo(E_EventType.InputSkill1, InputInfo.E_InputType.Down, info.key);
                }
                else //按下了鼠标
                {
                    Debug.Log("按键修改为：" + info.mouseID);
                    InputMgr.Instance.ChangeMouseInfo(E_EventType.InputSkill3, InputInfo.E_InputType.Down, info.mouseID);
                }
            });
        }
    }
}
```
