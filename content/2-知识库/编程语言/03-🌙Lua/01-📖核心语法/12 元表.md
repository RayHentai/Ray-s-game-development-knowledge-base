## 核心要点

**1. 元表的概念**
- 任何表变量都可以作为另一个表变量的元表
- 任何表变量都可以有自己的元表（爸爸）
- 当在子表中进行一些特定操作时，会执行元表中的内容

**2. 设置元表**
- `setmetatable`(子表，元表) 设置元表
- `getmetatable`(子表) 得到元表

**3. 元表的特定操作**
- `__tostring` 子表作为字符串的默认操作
- `__call` 子表作为函数的默认操作
- `__index` 子表找不到索引时会去寻找`__index`中的索引
- `__newindex` 子表为寻找不到的一个元素赋值时 会修改`__newindex`中的表
- 运算符重载

**4. raw**
- rawget 绕开 `__index` 获取值
- rawset 绕开 `__newindex` 设置值

**5.元表支持套娃**

## 代码演示

```lua
--元表
--元表概念
--任何表变量都可以作为另一个表变量的元表
--任何表变量都可以有自己的元表（爸爸）
--当子表中进行一些特定操作时会执行元表中的内容

--设置元表
meta = {}
myTable = {}
--设置元表函数
--参数一 子表
--参数二 元表
setmetatable(myTable, meta)

--特定操作
--__tostring
--当子表要被当作字符串使用时 会默认调用这个元表中的tostring方法
--有一个默认参数 代表子表自己
meta2 = {
	__tostring = function(t)
		return t.name
	end
}
myTable2 = {
	name = "子表"
}
setmetatable(myTable2, meta2)
print(myTable2) --子表

--__call
--当子表要被当作函数使用时 会默认调用这个元表中call的内容
--有一个默认参数 代表子表自己 第二个参数之后才代表传入的参数
meta3 = {
	__tostring = function(t)
		return t.name
	end,
	__call = function(a, b)
		print("元表中的方法")
		print(a)
		print(b)
	end
}
myTable3 = {
	name = "子表"
}
setmetatable(myTable3, meta3)
myTable3(1) --元表中的方法 子表 1

--运算符重载
meta4 = {
	__add = function(t1, t2) --当子表使用+运算符时会调用该方法
		return t1.age + t2.age
	end,
	__sub = function(t1, t2) --当子表使用-运算符时会调用该方法
		return t1.age - t2.age
	end,
	__mul = function(t1, t2) --当子表使用*运算符时会调用该方法
		return t1.age * t2.age
	end,
	__div = function(t1, t2) --当子表使用/运算符时会调用该方法
		return t1.age / t2.age
	end,
	__mod = function(t1, t2) --当子表使用%运算符时会调用该方法
		return t1.age % t2.age
	end,
	__pow = function(t1, t2) --当子表使用^运算符时会调用该方法
		return t1.age ^ t2.age
	end,
	__eq = function(t1, t2) --当子表使用==运算符时会调用该方法
		return t1.age == t2.age and true or false
	end,
	__lt = function(t1, t2) --当子表使用<运算符时会调用该方法
		return t1.age < t2.age and true or false
	end,
	__le = function(t1, t2) --当子表使用<=运算符时会调用该方法
		return t1.age <= t2.age and true or false
	end,
	__concat = function(t1, t2) --当子表使用..运算符时会调用该方法
		return t1.age .. t2.age
	end,
}
myTable4 = {age = 1}
myTable5 = {age = 2}
setmetatable(myTable4, meta4)
setmetatable(myTable5, meta4)
print(myTable4 + myTable5) --3
print(myTable4 - myTable5) ---1
print(myTable4 * myTable5) --2
print(myTable4 / myTable5) --0.5
print(myTable4 % myTable5) --1
print(myTable4 ^ myTable5) --1
--如果要用条件运算符的重载来比较两个对象 这两个对象的元表需要一致
print(myTable4 == myTable5) --false
print(myTable4 < myTable5) --true
print(myTable4 <= myTable5) --true
print(myTable4 .. myTable5) --12

--__index和__newindex
--当子表中 找不到某一个属性时 会到元表中__index指定的表去找索引
--__index 的赋值建议写在外面初始化 避免出问题
meta6 = {}
myTable6 = {
	
}
setmetatable(myTable6, meta6)
print(myTable6.age) --nil
meta6 = {
	__index = {age = 2}
}
setmetatable(myTable6, meta6)
print(myTable6.age) --2
print(rawget(myTable6, "age")) --获取自己身上的属性

meta6 = {
	age = 3
}
meta6.__index = meta6
setmetatable(myTable6, meta6)
print(myTable6.age) --3

--__index的套娃
meta7Father = {
	age = 7
}
meta7Father.__index = meta7Father
meta7 = {}
meta7.__index = meta7
myTable7 = {}

setmetatable(myTable7, meta7)
setmetatable(meta7, meta7Father)
print(myTable7.age) --7

--__newindex 当赋值时 如果赋值一个不存在的索引 
--那么会把这个值赋值到__newIndex所指的表中 不会修改自己
--__newindex 也支持套娃
meta8 = {}
meta8.__newindex = {}
myTable8 = {}
setmetatable(myTable8, meta8)
myTable8.age = 1
print(myTable8.age) --nil
print(meta8.__newindex.age) --1
rawset(myTable8, "age", 2) --忽略__newindex的设置 只会修改自己的变量
print(myTable8.age) --2

--获取元表
print(getmetatable(myTable8))

```
