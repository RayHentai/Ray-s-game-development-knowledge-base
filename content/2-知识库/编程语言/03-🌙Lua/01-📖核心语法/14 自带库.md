## 代码演示

```lua
--自带库
--时间
print(os.time()) --系统时间
print(os.time({year = 2020, month = 8, day = 14})) --自己传入时间
local nowTime = os.date("*t") --得到详细时间 是一个表
for k,v in pairs(nowTime) do --迭代器遍历可以得到具体信息
	print(k, v)
end
print(nowTime.hour) --得到当前时间相关的属性

--数学运算
print(math.abs(-11)) --11 绝对值
print(math.deg(math.pi)) --180 弧度转角度
print(math.cos(math.pi)) ---1 三角函数 传入弧度
print(math.floor(2.6)) --2 向下取整
print(math.ceil(5.2)) --6 向上取整
print(math.max(1,2)) --2 最大值
print(math.min(2,4)) --2 最小值
print(math.modf(1.2)) --1 0.2整数小数分离
print(math.pow(2,5)) --32 幂运算
--随机数
--先设置随机种子 随便传
math.randomseed(os.time())
print(math.random(100)) --第一个随机数受到随机数种子影响 种子变了才会变
print(math.random(100)) --第二个数会正常随机
print(math.sqrt(4)) --2 开方

--路径
print(package.path) --得到路径
package.path = package.path ..";C:\\" --改路径
print(package.path)
```
