## 需求分析

**游戏目标**：两个玩家（玩家和电脑）在棋盘上移动棋子，先到达终点的获胜。

**玩家操作**：按任意键掷骰子（随机1~6），根据骰子结果移动。

**特殊格子**：
- 普通格子：正常移动
- 幸运格子：可以选择换位置或轰炸对方
- 地雷格子：退后6格
- 暂停格子：暂停一回合
- 时空隧道：前进10格

**胜负条件**：抵达终点则获胜。

//场景枚举
 enum E_SceneType
 {
     kaiShi,
     youXi,
     jieShu,
 }
 //格子枚举
 enum E_GeZiType
{
    puTong,
    zanTing,
    zhaDan,
    shiKongSuiDao,
}
/// <summary>
/// 玩家枚举
/// </summary>
enum E_PlayerType
{
    player,
    computer,
}

//位置结构体
struct Vector
{
    public int x;
    public int y;
    public Vector(int x, int y)
    {
        this.x = x;
        this.y = y;
    }
}
//格子结构体
struct GeZi
{
    //格子类型
    public E_GeZiType type;
    //格子位置
    public Vector pos;

    public GeZi(int x, int y, E_GeZiType type)
    {
        pos.x = x;
        pos.y = y;
        this.type = type;
    }
    public void Draw()
    {
        Console.SetCursorPosition(pos.x, pos.y);
        switch (type)
        {
            case E_GeZiType.puTong:
                Console.ForegroundColor = ConsoleColor.White;
                Console.Write("□");
                break;
            case E_GeZiType.zanTing:
                Console.ForegroundColor = ConsoleColor.Blue;
                Console.Write("∥");
                break;
            case E_GeZiType.zhaDan:
                Console.ForegroundColor = ConsoleColor.Red;
                Console.Write("●");
                break;
            case E_GeZiType.shiKongSuiDao:
                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.Write("◆");
                break;
        }
    }
}
//地图结构体
struct Map
{
    //生成一个格子的数组
    public GeZi[] g;
    //格子初始化构造函数
    public Map(int x, int y, int num)//格子的 初始位置 数量
    {
        //初始化格子类型 位置
        g = new GeZi[num];
        Random r = new Random();
        int ramdomNum;
        //位置判断相关标识 x y 步长
        int indexX = 0;
        int indexY = 0;
        int stepNum = 2;
        for (int i = 0; i < num; i++)
        {
            ramdomNum = r.Next(0, 101);
            //首尾为普通格子 其他格子85%的概率为普通格子
            if (i == 0 || i == num - 1 || ramdomNum < 85)
            {
                //数组的格子类型 = 普通格子
                g[i].type = E_GeZiType.puTong;
            }
            else if (ramdomNum >= 85 && ramdomNum < 90)
            {
                g[i].type = E_GeZiType.zanTing;
            }
            else if (ramdomNum >= 90 && ramdomNum < 95)
            {
                g[i].type = E_GeZiType.zhaDan;
            }
            else
            {
                g[i].type = E_GeZiType.shiKongSuiDao;
            }
            //初始化格子位置
            g[i].pos = new Vector(x, y);
            //每次循环都要按照一定的规则去变化位置
            if (indexX == 10)
            {
                indexY++;
                y += 1;
                if (indexY == 2)
                {
                    indexX = 0;
                    indexY = 0;
                    stepNum = -stepNum;
                }
            }
            else
            {
                indexX++;
                x += stepNum;
            }
        }
    }
    //画格子的函数 
    public void Draw()
    {
        //执行遍历 把每个格子画一遍
        for (int i = 0; i < g.Length; i++)
        {
            g[i].Draw();
        }
    }
}
//玩家结构体
struct Player
{
    //玩家类型
    public E_PlayerType type;
    //玩家位置（地图数组的下标）
    public int nowIndex;
    //是否暂停的标识
    public bool isPause;
    public Player(int index, E_PlayerType type)
    {
        nowIndex = index;
        this.type = type;
        isPause = false;
    }
    public void Draw(Map mapInfo)
    {
        //得到格子信息
        GeZi geZi = mapInfo.g[nowIndex];
        //设置光标位置
        Console.SetCursorPosition(geZi.pos.x, geZi.pos.y);
        //绘制玩家
        switch (type)
        {
            case E_PlayerType.player:
                Console.ForegroundColor = ConsoleColor.Cyan;
                Console.Write("★");
                break;
            case E_PlayerType.computer:
                Console.ForegroundColor = ConsoleColor.Magenta;
                Console.Write("▲");
                break;
        }
    }
}

//设置控制台
static void ConsoleInit(int w, int h)
{
    Console.CursorVisible = false;
    Console.SetWindowSize(w, h);
    Console.SetBufferSize(w, h);
}

//开始场景
static void KaiShiScene(int w, int h, ref E_SceneType nowScene)
{
    Console.ForegroundColor = ConsoleColor.White;
    Console.SetCursorPosition(nowScene == E_SceneType.kaiShi ? w / 2 - 3 : w / 2 - 4, 6);
    Console.Write(nowScene == E_SceneType.kaiShi ? "飞行棋" : "游戏结束");
    //声明选项标识
    int selIndex = 0;
    bool isQuitWhile = false;
    //开始场景循环
    while (true)
    {
        //当前选项
        Console.ForegroundColor = selIndex == 0 ? ConsoleColor.Red : ConsoleColor.White;
        Console.SetCursorPosition(nowScene == E_SceneType.kaiShi ? w / 2 - 4 : w / 2 - 5, 10);
        Console.Write(nowScene == E_SceneType.kaiShi ? "开始游戏" : "回到主菜单");
        Console.ForegroundColor = selIndex == 1 ? ConsoleColor.Red : ConsoleColor.White;
        Console.SetCursorPosition(w / 2 - 4, 12);
        Console.Write("结束游戏");
        //检测输入
        switch (Console.ReadKey(true).Key)//Key就是一个枚举 
        {
            case ConsoleKey.W:
                selIndex--;
                if (selIndex < 0)
                {
                    selIndex = 0;
                }
                break;
            case ConsoleKey.S:
                selIndex++;
                if (selIndex > 1)
                {
                    selIndex = 1;
                }
                break;
            case ConsoleKey.J:
                if (selIndex == 0)
                {
                    nowScene = nowScene == E_SceneType.kaiShi ? E_SceneType.youXi : E_SceneType.kaiShi;
                    isQuitWhile = true;
                }
                else
                {
                    Environment.Exit(0);
                }
                break;
        }
        if (isQuitWhile)
        {
            break;
        }
    }
}

//画地图
static void DrawWall(int w, int h)
{
    for (int i = 0; i < w; i += 2)

        Console.ForegroundColor = ConsoleColor.Red;
        Console.SetCursorPosition(i, 0);
        Console.Write("■");
        Console.SetCursorPosition(i, h - 11);
        Console.Write("■");
        Console.SetCursorPosition(i, h - 6);
        Console.Write("■");
        Console.SetCursorPosition(i, h - 1);
        Console.Write("■");
    }
    for (int i = 0; i < h; i++)
    {
        Console.SetCursorPosition(0, i);
        Console.Write("■");
        Console.SetCursorPosition(w - 2, i);
        Console.Write("■");
    }
    Console.ForegroundColor = ConsoleColor.White;
    Console.SetCursorPosition(2, h - 10);
    Console.Write("□:普通格子");
    Console.ForegroundColor = ConsoleColor.Blue;
    Console.SetCursorPosition(2, h - 9);

//画玩家
static void DrawPlayer(Player player, Player computer, Map map)
{
    //重合时
    if (player.nowIndex == computer.nowIndex)
    {
        GeZi geZi = map.g[player.nowIndex];
        Console.SetCursorPosition(geZi.pos.x, geZi.pos.y);
        Console.ForegroundColor = ConsoleColor.DarkGreen;
        Console.Write("◎");
    }
    //不重合时
    else
    {
        player.Draw(map);
        computer.Draw(map);
    }
}

//清除信息框
static void ClearInfo(int w, int h)
{
    Console.SetCursorPosition(2, h - 5);
    Console.Write("                                        ");
    Console.SetCursorPosition(2, h - 4);
    Console.Write("                                        ");
    Console.SetCursorPosition(2, h - 3);
    Console.Write("                                        ");
    Console.SetCursorPosition(2, h - 2);
    Console.Write("                                        ");
}

/// <summary>
/// 扔色子函数
/// </summary>
/// <param name="w">窗口的宽</param>
/// <param name="h">窗口的高</param>
/// <param name="p">扔色子的对象</param>
/// <param name="map">地图信息</param>
/// <returns>代表没有结束</returns>
static bool ShaiZi(int w, int h, ref Player p, ref Player p2, Map map)
{
    //仍色子之前 判断 玩家是否暂停
    //擦除提示信息
    ClearInfo(w, h);
    //根据扔色子的玩家类型 改变打印颜色
    Console.ForegroundColor = p.type == E_PlayerType.player ? ConsoleColor.Cyan : ConsoleColor.Magenta;
    if (p.isPause)
    {
        Console.SetCursorPosition(2, h - 5);
        Console.Write("处于暂停点，{0}需要暂停一回合", p.type == E_PlayerType.player ? "你" : "电脑");
        Console.SetCursorPosition(2, h - 4);
        Console.Write("请按任意键，让{0}开始扔色子", p.type == E_PlayerType.player ? "电脑" : "你");
        //停止暂停
        p.isPause = false;
        return false;
    }
    Random r = new Random();
    int randomNum = r.Next(1, 7);
    //打印仍出点数信息
    Console.SetCursorPosition(2, h - 5);
    Console.Write("{0}扔出的点数为：{1}", p.type == E_PlayerType.player ? "你" : "电脑", randomNum);
    p.nowIndex += randomNum;
    //首先判断是否到终点
    if (p.nowIndex >= map.g.Length - 1)
    {
        p.nowIndex = map.g.Length - 1;
        Console.SetCursorPosition(2, h - 4);
        if (p.nowIndex == map.g.Length - 1)
        {
            Console.Write("恭喜你，你率先到达了终点");
            Console.SetCursorPosition(2, h - 3);
            Console.Write("请按任意键结束游戏");

        }
        else
        {
            Console.Write("很遗憾，电脑率先到达了终点");
            Console.SetCursorPosition(2, h - 3);
            Console.Write("请按任意键结束游戏");
        }
        return true;
    }
    //如果没有到终点 就判断当前对象 到了哪个格子
    else
    {
        GeZi geZi = map.g[p.nowIndex];
        switch (geZi.type)
        {
            case E_GeZiType.puTong:
                //没有变化
                Console.SetCursorPosition(2, h - 4);
                Console.Write("{0}到达了一个安全位置", p.type == E_PlayerType.player ? "你" : "电脑");
                Console.SetCursorPosition(2, h - 3);
                Console.Write("请按任意键，让{0}开始扔色子", p.type == E_PlayerType.player ? "电脑" : "你");
                break;
            case E_GeZiType.zanTing:
                //暂停
                //加一个标识才能知道 下一回合是不是不能扔色子
                //下一回合要暂停
                Console.SetCursorPosition(2, h - 4);
                Console.Write("{0}踩到了暂停点，暂停一个回合", p.type == E_PlayerType.player ? "你" : "电脑");
                Console.SetCursorPosition(2, h - 3);
                Console.Write("请按任意键，让{0}开始扔色子", p.type == E_PlayerType.player ? "电脑" : "你");
                p.isPause = true;
                break;
            case E_GeZiType.zhaDan:
                //退格
                p.nowIndex -= 5;
                if (p.nowIndex < 0)
                {
                    //不能比起点小
                    p.nowIndex = 0;
                }
                Console.SetCursorPosition(2, h - 4);
                Console.Write("{0}踩到了炸弹，退后5格", p.type == E_PlayerType.player ? "你" : "电脑");
                Console.SetCursorPosition(2, h - 3);
                Console.Write("请按任意键，让{0}开始扔色子", p.type == E_PlayerType.player ? "你" : "电脑");
                break;
            case E_GeZiType.shiKongSuiDao:
                //随机 
                Console.SetCursorPosition(2, h - 4);
                Console.Write("{0}踩到了时空隧道", p.type == E_PlayerType.player ? "你" : "电脑");
                randomNum = r.Next(1, 90);
                //触发倒退
                if (randomNum < 30)
                {
                    p.nowIndex -= 5;
                    if (p.nowIndex < 0)
                    {
                        p.nowIndex = 0;
                    }
                    Console.SetCursorPosition(2, h - 3);
                    Console.Write("触发倒退5格");
                }
                //触发暂停
                else if (randomNum < 60)
                {
                    p.isPause = true;
                    Console.SetCursorPosition(2, h - 3);
                    Console.Write("触发暂停一回合");
                }
                //触发交换
                else
                {
                    int temp = p.nowIndex;
                    p.nowIndex = p2.nowIndex;
                    p2.nowIndex = temp;
                    Console.SetCursorPosition(2, h - 3);
                    Console.Write("倒霉！触发交换");
                }

                Console.SetCursorPosition(2, h - 2);
                Console.Write("请按任意键，让{0}开始扔色子", p.type == E_PlayerType.player ? "你" : "电脑");
                break;
        }
    }
    //默认没有结束
    return false;
}

//玩家移动
static bool PlayerRandoMove(int w, int h, ref Player p, ref Player p2, Map m, ref E_SceneType nowScene)//返回bool值退出循环
{
    //玩家仍色子逻辑
    //检测输入
    Console.ReadKey(true);
    //扔色子
    bool isGameOver = ShaiZi(w, h, ref p, ref p2, m);
    //绘制地图
    m.Draw();
    //绘制玩家
    DrawPlayer(p, p2, m);
    //判断是否结束游戏
    if (isGameOver)
    {
        //卡住程序 让玩家按任意键
        Console.ReadKey(true);
        //改变场景ID
        nowScene = E_SceneType.jieShu;
        //跳出循环
    }
    return isGameOver;
}
