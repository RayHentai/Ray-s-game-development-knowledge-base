## 需求分析

**游戏目标**：击倒Boss营救公主

**玩家操作**：WASD移动J互动

**胜负条件**：击倒Boss营救公主取得胜利

## 功能拆解

> 把整个项目拆成独立的功能模块，每个功能是一个可以单独实现的单元

- 功能一：场景切换，不同的场景执行不同的逻辑
- 功能二：输入检测，检测不同的输入执行不同的逻辑
- 功能三：玩家的移动
- 功能四：地图、场景和NPC的绘制
- 功能五：自动战斗显示战斗信息

## 完整工程


```csharp
//1、控制台设置
int w = 60, h = 30;
Console.SetWindowSize(w, h);
Console.SetBufferSize(w, h);
Console.CursorVisible = false;

//2、游戏场景设置
int nowSceneID = 1;
string gameOver = "";

while (true)//三个不同场景的循环
{
    switch (nowSceneID)
    {
	    //3、开始游戏场景
        case 1:
            Console.Clear();
	        //3.1、开始游戏场景——标题
            Console.SetCursorPosition(w / 2 - 4, 5);
            Console.Write("营救公主");
            
            //3.2、开始游戏场景——标识
            int nowSelIndex = 0;
            bool isQuitWhile = false;
            while (true)
            {
                //3.3、开始游戏场景——选项
                Console.SetCursorPosition(w / 2 - 4, 11);
                Console.ForegroundColor = nowSelIndex == 0 ? ConsoleColor.Red : ConsoleColor.White;
                Console.Write("开始游戏");
                Console.SetCursorPosition(w / 2 - 4, 13);
                Console.ForegroundColor = nowSelIndex == 1 ? ConsoleColor.Red : ConsoleColor.White;
                Console.Write("退出游戏");
                
                //3.4、开始游戏场景——输入检测
                char c = Console.ReadKey(true).KeyChar;
                switch (c)
                {
                    case 'W':
                    case 'w':
                        --nowSelIndex;
                        if (nowSelIndex < 0)
                            nowSelIndex = 0;
                        break;
                    case 'S':
                    case 's':
                        ++nowSelIndex;
                        if (nowSelIndex > 1)
                            nowSelIndex = 1;
                            break;
                    case 'J':
                    case 'j':
                        if (nowSelIndex == 0)
                        {
                            nowSceneID = 2;
                            isQuitWhile = true;
                        }
                        else
                        {
                            Environment.Exit(0);
                        }
                        break;
                }
                if (isQuitWhile)
                    break;
            }
                    break;
        //3、游戏场景
        case 2:
            Console.Clear();
            //3.1、游戏场景——不变的红墙
            Console.ForegroundColor = ConsoleColor.Red;
            for (int i = 0; i <= w - 2; i += 2)
            {
                Console.SetCursorPosition(i, 0);
                Console.Write("■");
                Console.SetCursorPosition(i, h - 1);
                Console.Write("■");
                Console.SetCursorPosition(i, h - 6);
                Console.Write("■");
            }
            for (int i = 0; i <= h - 1; ++i)
            {
                Console.SetCursorPosition(0, i);
                Console.Write("■");
                Console.SetCursorPosition(w - 2, i);
                Console.Write("■");
            }
            
            //3.2、游戏场景——boss相关
            int bossX = w / 2, bossY = h / 2;
            int bossminAtk = 7;
            int bossMaxAkt = 14;
            int bossHp = 100;
            string bossIcon = "■";
            ConsoleColor bossColor = ConsoleColor.Green;//申明一个颜色变量
            
            //3.3、游戏场景——玩家相关
            int playerX = 4, playerY = 4;
            int playerMinAtk = 8;
            int playerMaxAtk = 15;
            int playerHp = 100;
            string playerIcon = "●";
            ConsoleColor playerColor = ConsoleColor.Yellow;
            
            //3.4、游戏场景——公主相关
            int gongZhuX = w / 2, gongZhuY = 5;
            string gongZhuIcon = "★";
            ConsoleColor gongZhuColor = ConsoleColor.Blue;
            
            //3.5、游戏场景——标识
            bool isFight = false;
            bool isOver = false;
            char playerInput;
            while (true)//在外层就把变量申明好 方便后续修改 
            {
				//3.6、游戏场景——战斗相关
                if (bossHp > 0)
                {
                    Console.SetCursorPosition(bossX, bossY);
                    Console.ForegroundColor = bossColor;
                    Console.Write(bossIcon);
                }
                else
                {
                    Console.SetCursorPosition(gongZhuX, gongZhuY);
                    Console.ForegroundColor = gongZhuColor;
                    Console.Write(gongZhuIcon);
                }
                Console.ForegroundColor = playerColor;
                Console.SetCursorPosition(playerX, playerY);
                Console.Write(playerIcon);
                playerInput = Console.ReadKey(true).KeyChar;
                Console.SetCursorPosition(playerX, playerY);
                Console.Write(" ");
                if (isFight)//是否是战斗状态
                {
                    Random r = new Random();
                    int playerAtk;
                    int bossAtk;
                    if (playerInput == 'J' || playerInput == 'j')
                    {
                        if (playerHp <= 0)
                        {
                            nowSceneID = 3;
                            break;
                        }
                        else if (bossHp <= 0)
                        {
                            Console.SetCursorPosition(bossX, bossY);
                            Console.Write(" ");
                            isFight = false;
                        }
                        playerAtk = r.Next(playerMinAtk, playerMaxAtk);
                        bossHp -= playerAtk;
                        Console.ForegroundColor = bossColor;
                        Console.SetCursorPosition(2, h - 4);
                        Console.Write("                                       ");
                        Console.SetCursorPosition(2, h - 4);
                        Console.Write("你本次造成{0}伤害，boss剩余血量{1}", playerAtk, bossHp);
                        if (bossHp > 0)
                        {
                            bossAtk = r.Next(bossminAtk, bossMaxAkt);
                            playerHp -= bossAtk;
                            Console.SetCursorPosition(2, h - 3);
                            Console.Write("                                   ");
                            if (playerHp <= 0)
                            {
                                Console.SetCursorPosition(2, h - 3);
                                Console.Write("很遗憾，你未能通过boss的试炼");
                                gameOver = "游戏失败";
                            }
                            else
                            {
                                Console.ForegroundColor = playerColor;
                                Console.SetCursorPosition(2, h - 3);
                                Console.Write("boss对你造成{0}伤害，你剩余血量{1}", bossAtk, playerHp);
                            }
                        }
                        else
                        {
                            Console.SetCursorPosition(2, h - 5);
                            Console.Write("                                             ");
                            Console.SetCursorPosition(2, h - 4);
                            Console.Write("                                             ");
                            Console.SetCursorPosition(2, h - 3);
                            Console.Write("                                             ");
                            Console.ForegroundColor = ConsoleColor.White;
                            Console.SetCursorPosition(2, h - 5);
                            Console.Write("战斗结束，boss已被打倒，快去拯救公主吧");
                            Console.SetCursorPosition(2, h - 4);
                            Console.Write("前往公主身边按j继续");
                            gameOver = "恭喜通关";
                        }
                    }
                }
                else
                {
                    switch (playerInput)
                    {
                        case 'W':
                        case 'w':
                            --playerY;
                            if (playerY < 1)
                            {
                                ++playerY;
                            }
                            else if (playerX == bossX && 
                            playerY == bossY && bossHp > 0 || 
                            playerX == gongZhuX && 
                            playerY == gongZhuY && bossHp < 0)
                                ++playerY;
                            break;
                        case 'A':
                        case 'a':
                            playerX -= 2;
                            if (playerX < 2)
	                            playerX += 2;
                            else if (playerX == bossX && 
                            playerY == bossY && bossHp > 0 ||
                            playerX == gongZhuX && 
                            playerY == gongZhuY && bossHp < 0)
                                playerX += 2;
                            break;
                        case 'S':
                        case 's':
                            ++playerY;
                            if (playerY > h - 7)
                                playerY = h - 7;
                            else if (playerX == bossX &&
                             playerY == bossY && bossHp > 0 ||
                             playerX == gongZhuX && 
                             playerY == gongZhuY && bossHp < 0)
                                --playerY;
                            break;
                        case 'D':
                        case 'd':
                            playerX += 2;
                            if (playerX > w - 4)
                                playerX = w - 4;
                            else if (playerX == bossX && 
                            playerY == bossY && bossHp > 0 ||
                            playerX == gongZhuX && 
                            playerY == gongZhuY && bossHp < 0)
                                playerX -= 2;
                            break;
                        case 'J':
                        case 'j':
                            if ((playerX == bossX && 
                            playerY == bossY - 1 ||
                            playerX == bossX && playerY == bossY + 1 || 
                            playerY == bossY && playerX == bossX - 2 ||
                            playerY == bossY && playerX == bossX + 2) && 
                            bossHp > 0)//判断玩家位置 只有玩家在boss 旁边按J才能继续
                            {
                                Console.ForegroundColor = ConsoleColor.White;
                                Console.SetCursorPosition(2, h - 5);
                                Console.Write("开始和boss战斗，按J继续");
                                Console.ForegroundColor = ConsoleColor.Yellow;
                                Console.SetCursorPosition(2, h - 4);
                                Console.Write("玩家当前血量为{0}", playerHp);
                                Console.ForegroundColor = ConsoleColor.Green;
                                Console.SetCursorPosition(2, h - 3);
                                Console.Write("boss当前血量为{0}", bossHp);
                                isFight = true;
                            }
                            else if ((playerX == gongZhuX && 
                            playerY == gongZhuY - 1 ||
                            playerX == gongZhuX && playerY == gongZhuY + 1 ||
                            playerY == gongZhuY && playerX == gongZhuX - 2 ||
                            playerY == gongZhuY && playerX == gongZhuX + 2) &&
                            bossHp < 0)
                            {
                                nowSceneID = 3;
                                isOver = true;
                            }
                            break;
                    }
                    if (isOver)
                        break;
                }
            }
            break;
        //4、结束场景
        case 3:
            //4.1、结束场景——标题
            Console.Clear();
            Console.ForegroundColor = ConsoleColor.White;
            Console.SetCursorPosition(w / 2 - 4, 5);
            Console.Write("GameOver");
            Console.ForegroundColor = ConsoleColor.Green;
            Console.SetCursorPosition(w / 2 - 4, 7);
            Console.Write(gameOver);

            //4.2、结束场景——标识
            int nowSelOverIndex = 0;
            bool isQuitEndWhile = false;
            while (true)
            {
                //4.3、结束场景——选项
                Console.SetCursorPosition(w / 2 - 6, 11);
                Console.ForegroundColor = nowSelOverIndex == 0 ? ConsoleColor.Red : Console.ForegroundColor = ConsoleColor.White;
                Console.Write("回到开始界面");
                Console.SetCursorPosition(w / 2 - 4, 13);
                Console.ForegroundColor = nowSelOverIndex == 1 ? ConsoleColor.Red : Console.ForegroundColor = ConsoleColor.White;
                Console.Write("结束游戏");

                //4.4、结束场景——检测玩家输入
                playerInput = Console.ReadKey(true).KeyChar;
                switch (playerInput)
                {
                    case 'W':
                    case 'w':
                        --nowSelOverIndex;
                        if (nowSelOverIndex < 0)
                            nowSelOverIndex = 0;
                        break;
                    case 'S':
                    case 's':
                        ++nowSelOverIndex;
                        if (nowSelOverIndex > 1)
                            nowSelOverIndex = 1;
                        break;
	                case 'J':
                    case 'j':
                        if (nowSelOverIndex == 1)
                            Environment.Exit(0);
                        else
                        {
                            nowSceneID = 1;
                            isQuitEndWhile = true;
                        }
                        break;
                }
                if (isQuitEndWhile)
                        break;
            }
            break;
                }
            }
```
