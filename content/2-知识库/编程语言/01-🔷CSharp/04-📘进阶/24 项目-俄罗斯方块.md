```csharp
// 位置结构体
struct Position
{
    public int x;
    public int y;
    public Position(int x, int y)
    {
        this.x = x;
        this.y = y;
    }
    // 重载运算符 判断位置是否相等 [[12 运算符重载#^3c3d66]]
    public static bool operator ==(Position p1, Position p2)
    {
        if (p1.x == p2.x && p1.y == p2.y)
            return true;
        return false;
    }
    public static bool operator !=(Position p1, Position p2)
    {
        if (p1.x == p2.x && p1.y == p2.y)
            return true;
        return false;
    }
    // 重载运算符 相加两个位置
    public static Position operator +(Position p1, Position p2)
    {
        Position pos = new Position(p1.x + p2.x, p1.y + p2.y);
        return pos;
    }
}
```

```csharp
    {
        //while (isRunning)
        //{
            if (Console.KeyAvailable)
            {
                lock (blockWorker) //[[18 多线程#^256ccf]]
                {
                    switch (Console.ReadKey(true).Key)
                    {
                        case ConsoleKey.LeftArrow:
                            if (blockWorker.CanChange(E_ChangeType.left, map))
                                blockWorker.Change(E_ChangeType.left);
                            break;
                        case ConsoleKey.RightArrow:
                            if (blockWorker.CanChange(E_ChangeType.right, map))
                                blockWorker.Change(E_ChangeType.right);
                            break;
                        case ConsoleKey.A:
                            if (blockWorker.CanMove(E_ChangeType.left, map))
                                blockWorker.MoveRl(E_ChangeType.left);
                            break;
                        case ConsoleKey.D:
                            if (blockWorker.CanMove(E_ChangeType.right, map))
                                blockWorker.MoveRl(E_ChangeType.right);
                            break;
                            //按s快速向下移动
                        case ConsoleKey.S:
                            if (blockWorker.CanAutoMove(map))
                                blockWorker.AutoMove();
                            break;
                    }
                }
            }
        //}
    }
    //线程置空
    public void StopThread()
    {
        //isRunning = false;
        //inputThread = null;
        InputThread.Instance.inputEvent -= CheckInputThread;//[[13 事件#^b936e3]]
    }
    public void UpData()
    {
        lock (blockWorker)
        {
            map.Draw();
            str = string.Format("分数：{0}", map.score);
            Console.ForegroundColor = ConsoleColor.White;
            Console.SetCursorPosition(Game.w / 2 - str.Length / 2, Game.h - 3);

            //检测输入
            //如果判断能变方法为true 才能变
            if (blockWorker.CanAutoMove(map))
                blockWorker.AutoMove();
        }
        //这个锁了输入还是会有延迟

    }
```

```csharp
        ClearDraw();
        switch (type)
        {
            case E_ChangeType.left:
                --nowBlockIndex;
                //下标等于0 转到下标3
                if ( nowBlockIndex < 0 )
                    nowBlockIndex = nowBlockInfo.Count - 1;
                break;
            case E_ChangeType.right:
                ++nowBlockIndex;
                //下标等于3 转到下标0
                if (nowBlockIndex >= nowBlockInfo.Count)
                    nowBlockIndex = 0;
                break;
        }
        //得到索引信息是为了设置另外三个小方块
        Position[] pos = nowBlockInfo[nowBlockIndex];
        for (int i = 0; i < pos.Length; i++) 
        {
            blocks[i + 1].pos = blocks[0].pos + pos[i];
        }
        //变之后绘制
        Draw();
    }
    //判断能不能变
    public bool CanChange(E_ChangeType type, Map map)
    {
        //先用临时变量模拟变化
        //直接用成员变量 会真实变化
        int tempIndex = nowBlockIndex;
        switch (type)
        {
            case E_ChangeType.left:
                --tempIndex;
                //下标等于0 转到下标3
                if (tempIndex < 0)
                    tempIndex = nowBlockInfo.Count - 1;
                break;
            case E_ChangeType.right:
                ++tempIndex;
                //下标等于3 转到下标0
                if (tempIndex >= nowBlockInfo.Count)
                    tempIndex = 0;
                break;
        }
        //取出位置信息
        Position[] nowPos = nowBlockInfo[tempIndex];
        Position tempPos;
        //判断边界
        for (int i = 0; i < nowPos.Length; i++)
        {
            tempPos = blocks[0].pos + nowPos[i];
            if (tempPos.x < 2 || tempPos.x >= Game.w - 2 || tempPos.y > map.h)
                return false;
        }
        //判断动态方块
        for (int i = 0; i < nowPos.Length; i++)
        {
            tempPos = blocks[0].pos + nowPos[i];
            for (int j = 0; j < map.dynamicWalls.Count; j++) 
            {
                if ( tempPos == map.dynamicWalls[j].pos )
                    return false;
            }
        }
        return true;
    }
    //方块左右移动
    public void MoveRl(E_ChangeType type)
    {
        //移动之前擦除
        ClearDraw();
        //根据传入的位置 决定左动还是右动 
        //左动就是x整体-2 右动就是x整体+2
        //实例化一个位置变量
        Position pos = new Position(type == E_ChangeType.left ? -2 : 2, 0);
        for (int i = 0; i < blocks.Count; i++) 
        {
            blocks[i].pos += pos;
        }
        //移动之后绘制
        Draw();
    }
    //判断能不能移动
    public bool CanMove(E_ChangeType type, Map map)
    {
        Position pos = new Position(type == E_ChangeType.left ? -2 : 2, 0);
        Position tempPos;
        //判断地图边界
        for (int i = 0; i < blocks.Count; i++)
        {
            tempPos = blocks[i].pos + pos;
            if (tempPos.x < 2 || tempPos.x >= Game.w - 2)
                return false;
        }
        for (int i = 0; i < blocks.Count; i++)
        {
            tempPos = blocks[i].pos + pos;
            for (int j = 0; j < map.dynamicWalls.Count; j++) 
            {
                if (tempPos == map.dynamicWalls[j].pos)
                    return false;
            }
        }
        return true;
    }
    //方块自动移动
    public void AutoMove()
    {
        ClearDraw();
        Position pos = new Position(0, 1);
        for (int i = 0; i < blocks.Count; i++) 
        {
            blocks[i].pos += pos;
        }
        Draw();
    }
    //判断能不能自动移动
    public bool CanAutoMove(Map map)
    {
        Position pos = new Position(0, 1);
        Position tempPos;
        for (int i = 0; i < blocks.Count; i++)
        {
            tempPos = blocks[i].pos + pos;
            if (tempPos.y >= map.h)
            {
                //停下来 给与动态方块
                map.AddWalls(blocks);
                //创建新方块
                randomCreateBlock();
                return false;
            }
        }
        for (int i = 0; i < blocks.Count; i++)
        {
            tempPos = blocks[i].pos + pos;
            for (int j = 0; j < map.dynamicWalls.Count; j++)
            {
                if (tempPos == map.dynamicWalls[j].pos)
                {
                    //停下来 给与动态方块
                    map.AddWalls(blocks);
                    //创建新方块
                    randomCreateBlock();
                    return false;
                }
                    
            }
        }
        return true;
    }
    //擦除的方法
    public void ClearDraw()
    {
        for (int i = 0; i < blocks.Count; i++)
        {
            blocks[i].ClearDraw();
        }
    }

    public void Draw()
    {
        for (int i = 0; i < blocks.Count; i++)

            blocks[i].Draw();
        }
            
    }

    public void randomCreateBlock()
    {
        //随机一种类型 
        Random r = new Random();
        E_DrawType type = (E_DrawType)r.Next(1, 8);
        //实例化方块 确定方块类型
        blocks = new List<DrawObject>()
        {
            new DrawObject(type),
            new DrawObject(type),
            new DrawObject(type),
            new DrawObject(type),
        };
        //初始化原点方块位置
        blocks[0].pos = new Position(24, -5);
        //其他三个方块的位置
        //得到这一组方块的信息
        nowBlockInfo = blockInfoDic[type];
        //随机方块方向
        nowBlockIndex = r.Next(0, nowBlockInfo.Count);
        //取出位置信息数组
        Position[] pos = nowBlockInfo[nowBlockIndex];
        //让原点方块 加上位置信息数组中记录的相对坐标
        for (int i = 0; i < pos.Length; i++)
        {
            blocks[i + 1].pos = blocks[0].pos + pos[i];
        }
    }
```

#### 地图类

```csharp
class Map
{
    //List必须new 才能往里面装东西
    //固定墙壁
    private List<DrawObject> walls = new List<DrawObject>();
    //动态墙壁
    public List<DrawObject> dynamicWalls = new List<DrawObject>();
    //为了快速获得地图边界 
    private int w;
    public int h;
    //记录层数的数组
    private int[] recordInfo;
    private GameScene nowGameScene;
    //分数
    public int score;
    public Map(GameScene nowGameScene) 
    {
        score = 0;
        this.nowGameScene = nowGameScene;
        //方便外部得到地图 高的 边界信息 直接在此记录 避免多次修改代码
        h = Game.h - 6;
        w = 0;
        for (int i = 0; i < Game.w; i += 2) 
        {
            walls.Add(new DrawObject(E_DrawType.wall, i, h));
            ++w;
        }
        //记录墙壁个数
        w -= 2;
        for (int i = 0; i < h; i++) 
        {
            walls.Add(new DrawObject(E_DrawType.wall, 0, i));
            walls.Add(new DrawObject(E_DrawType.wall, Game.w - 2, i));
        }
        //初始化记录层数的数组 一共有Game.h - 6 层
        recordInfo = new int[h];
    }
    //绘制
    public void Draw()
    {
        //绘制静态墙壁
        for (int i = 0; i < walls.Count; i++) 
        {
            walls[i].Draw();
        }
        //绘制动态墙壁
        for(int i = 0;i < dynamicWalls.Count; i++)
        {
            dynamicWalls[i].Draw();
        }
    }
    //清除动态墙壁
    public void ClearDraw() 
    {
        for (int i = 0; i < dynamicWalls.Count; i++)
        {
            dynamicWalls[i].ClearDraw();
        }
    }
    // 提供给外部添加动态方块的方法
    public void AddWalls(List<DrawObject> walls)
    {
        for (int i = 0; i < walls.Count; i++) 
        {
            //传递方块进来时 将类型改为墙壁
            walls[i].ChangeType(E_DrawType.wall);
            //添加
            dynamicWalls.Add(walls[i]);
            //动态墙壁满了 就结束游戏
            if (walls[i].pos.y <= 0)
            {
                Game.ChangeScene(E_SceneType.end);
                nowGameScene.StopThread();
                return;
            }
            //新添加的方块 y 所在行  数组 + 1
            recordInfo[h - 1 - walls[i].pos.y] += 1;
        }
        //清除
        ClearDraw();
        //检查
        CheckClear();
        //绘制
        Draw();
    }
    //检测塌层
    public void CheckClear()
    {
        List<DrawObject> tempBlocks = new List<DrawObject>();
        //判断这一行是否满
        //遍历数组 检测数组里面存的数 是不是w - 2
        for (int i = 0; i < recordInfo.Length; i++) 
        {
            if (recordInfo[i] == w)
            {
                for (int j = 0; j < dynamicWalls.Count; j++) 
                {
                    //1 移除这一行的所有小方块
                    if (i == h - 1 - dynamicWalls[j].pos.y)
                    {
                        tempBlocks.Add(dynamicWalls[j]);
                    }
                    else if (i < h - 1 - dynamicWalls[j].pos.y)
                    {
                        //2 这一行之上所有小方块下移一个单位
                        ++dynamicWalls[j].pos.y;
                    }
                }
                //移除方块
                for (int j = 0; j < tempBlocks.Count; j++)
                {
                    dynamicWalls.Remove(tempBlocks[j]);
                    score += 10;
                }
                for (int j = i; j < recordInfo.Length - 1; j++) 
                {
                    //3 记录小方块数量的数组从上到下移动
                    recordInfo[j] = recordInfo[j + 1];
                }
                //最后一行需要置空
                recordInfo[recordInfo.Length - 1] = 0;
                //如果是连续消除多行的情况下 第1层 在第0层满的 消除之后 会变成 第0层 
                //最外层计数的i会从0 变成 1 就不会遍历第0层 如果这时的第0层是满的 就永远不会消除
                //这里用 break 和递归解决这个问题
                CheckClear();
                break;
            }
        }
    }
}
```
