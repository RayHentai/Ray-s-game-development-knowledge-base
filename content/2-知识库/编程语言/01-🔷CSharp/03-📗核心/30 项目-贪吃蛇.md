## 需求分析

**游戏目标：** 控制蛇在地图上移动，吃食物增长身体，碰到边界或自身则游戏结束。

**核心规则：**
- 蛇持续向当前方向移动（自动）
- 吃到食物：身体增长，在新位置生成食物
- 碰到边界或蛇身：游戏结束

```csharp

E_MoveDir up / down / left / right 移动方向
E_SneakBodyType head / body 蛇身体类型

【接口】
ISceneUpdata UpData() 场景帧更新行为
IDraw Draw() 绘制行为

【结构体】
Position x, y 位置结构体
 + == / != 运算符重载 判断两个位置是否重合

【抽象基类】
BeginOrEndBaseScene : ISceneUpdata
│  strTitle, strOne, nowSelIndex 标题、选项、当前选中索引
│  UpData() ← 实现接口（处理 WS 键选择 + J 键确认）
│  EnterJDoSomething() ← 抽象方法（由子类决定按 J 后的逻辑）
├── BeginScene 构造：标题"贪吃蛇"，选项"开始游戏"
│   └─ EnterJDoSomething() 选0 → 切换 Game 场景 / 选1 → 退出程序
└── EndScene 构造：标题"游戏结束"，选项"回到开始页面"
    └─ EnterJDoSomething() 选0 → 切换 Begin 场景 / 选1 → 退出程序

GameObject : IDraw
│  pos : Position 游戏对象位置
│  Draw()  ← 抽象方法（由子类实现具体绘制）
├── SnakeBody type : E_SneakBodyType
│   └─ Draw() head → ●黄色 / body → ◎绿色
├── Food ◆青色
│   └─ Draw() / randomPos(snake) 避开蛇身随机生成位置（递归）
└── Wall ■红色
    └─ Draw()

【普通类】
Map : IDraw
│  walls : Wall[] 存储所有墙体对象
│  Map() 构造函数：初始化上下左右四边墙体数组
└─ Draw() 遍历 walls 数组逐个调用 Draw()

Snake : IDraw
│  body : SnakeBody[200] 蛇的身体数组（固定200容量）
│  dir  : E_MoveDir 当前移动方向
│  nowNum : int 当前实际身体长度
│  Draw() 遍历 nowNum 个节点逐个绘制
│  Move() 擦除末尾 → 身体整体前移 → 蛇头按方向移动
│  ChangeDir(dir) 转向（禁止反向移动和重复方向）
│  CheckEnd(map) 检测撞墙或咬到自身 → 返回 bool
│  checkSamePos(p) 判断传入位置是否与蛇身任意节重合
│  checkEatFood(food) 蛇头与食物位置重合 → 随机食物位置 → AddBody()
└─ AddBody() 在末尾追加一节 body 类型的 SnakeBody

GameScene : ISceneUpdata
│  map, snake, food 三个核心游戏对象
│  updataIndex 降速计数器
│  GameScene() 构造函数：初始化 map / snake(40,10) / food(snake)
└─ UpData() 主循环逻辑：
                        ① 每 6000 次执行一次：绘制地图→绘制食物→蛇移动→绘制蛇
                           → CheckEnd (结束则切场景) → checkEatFood
                        ② Console.KeyAvailable：有输入才读取 → 调用 ChangeDir

Game
│  const w = 80, h = 20 游戏窗口宽高（常量，方便全局引用）
│  static nowScene : ISceneUpdata 当前运行场景（静态，方便跨类访问）
│  Game() 控制台初始化（隐藏光标/设置窗口和缓冲区大小）→ ChangeScene(begin)
│  Start() 主循环：while(true) → nowScene.UpData()
└─ static ChangeScene(E_SceneType)  Console.Clear() → new 对应场景赋值给 nowScene

Program（入口）
└─ new Game().Start()
```

```csharp
struct Position
{
    public int x;
    public int y;

    public Position(int x, int y)
    {
        this.x = x;
        this.y = y;
    }
    //贪吃蛇中肯定会存在位置的比较
    //各个游戏对象都会去比较位置是不是重合
    //重载运算符 [[12 运算符重载]]
    public static bool operator ==(Position p, Position p2)
    {
        if (p.x == p2.x && p.y == p2.y)
        {
            return true;
        }
        else
        {
            return false;
        }
    }
    public static bool operator !=(Position p, Position p2)
    {
        if (p.x == p2.x && p.y == p2.y)
        {
            return false;
        }
        else
        {
            return true;
        }
    }
}
```

```csharp
//开始场景和结束场景的基类
//因为这个类不会被实例化 所以可以作为一个抽象类使用
abstract class BeginOrEndBaseScene : ISceneUpdata
{
    //标题
    public string strTitle;
    //第一个选项
    public string strOne;
    //选项标识 默认等于0
    public int nowSelIndex = 0;

    //声明一个按J键后执行操作的抽象函数
    public abstract void EnterJDoSomething();
    //开始和结束场景的游戏逻辑
    public void UpData()
    {
        //输入颜色设置成白色 设置光标
        Console.ForegroundColor = ConsoleColor.White;
        Console.SetCursorPosition(Game.w / 2 - strTitle.Length, 5);
        //显示标题
        Console.Write(strTitle);
        //显示选项
        Console.ForegroundColor = nowSelIndex == 0 ? ConsoleColor.Red : ConsoleColor.White;
        Console.SetCursorPosition(Game.w / 2 - strOne.Length, 8);
        Console.Write(strOne);
        Console.ForegroundColor = nowSelIndex == 1 ? ConsoleColor.Red : ConsoleColor.White;
        Console.SetCursorPosition(Game.w / 2 - 4, 10);
        Console.Write("结束游戏");
        //检测输入
        switch (Console.ReadKey(true).Key)
        {
            case ConsoleKey.W:
                --nowSelIndex;
                if (nowSelIndex < 0)
                {
                    nowSelIndex = 0;
                }
                break;
            case ConsoleKey.S:
                ++nowSelIndex;
                if (nowSelIndex > 1)
                {
                    nowSelIndex = 1;
                }
                break;
            case ConsoleKey.J:
                //执行抽象方法
                EnterJDoSomething();
                break;
        }
    }
}

//开始场景
class BeginScene : BeginOrEndBaseScene
{
    //声明无参构造函数
    public BeginScene()
    {
        strTitle = "贪吃蛇";
        strOne = "开始游戏";
    }
    public override void EnterJDoSomething()
    {
        if (nowSelIndex == 0)
        {
            Game.ChangeScene(E_SceneType.game);
        }
        else
        {
            Environment.Exit(0);
        }
    }
}

//结束场景
class EndScene : BeginOrEndBaseScene
{
    //与开始场景逻辑完全一致 修改变量即可
    public EndScene()
    {
        strTitle = "游戏结束";
        strOne = "回到开始页面";
    }
    public override void EnterJDoSomething()
    {
        if (nowSelIndex == 0)
        {
            Game.ChangeScene(E_SceneType.begin);
        }
        else
        {
            Environment.Exit(0);
        }
    }
}
```

```csharp
//游戏对象基类
//因为游戏对象也是不会new的 所以可以修饰为抽象类
abstract class GameObject : IDraw
{
    //位置
    public Position pos;
    //继承接口的抽象类可以不实现抽象函数 由继承抽象类的子类实现
    abstract public void Draw();
}

//食物类
internal class Food : GameObject
{
    public Food(Snake snake)
    {
        //只要初始化就传进来蛇的坐标 执行随机位置
        randomPos(snake);
    }
    
    public override void Draw()
    {
        Console.ForegroundColor = ConsoleColor.Cyan;
        Console.SetCursorPosition(pos.x, pos.y);
        Console.Write("◆");
    }
    //随机位置的行为 和蛇头有关系 后面再处理
    public void randomPos(Snake snake)
    {
        //随机位置
        Random r = new Random();
        //x一定是偶数 因为一个字符占两个位置
        pos.x = r.Next(1, Game.w / 2 - 1) * 2;
        pos.y = r.Next(1, Game.h - 2);
        //得到蛇的位置
        if (snake.checkSamePos(pos))
        {
            //递归 如果不进循环继续判断 [[11 递归函数]]
            randomPos(snake);
        }
    }
}

//蛇身体类
enum E_SneakBodyType
{
    head,
    body,
}
internal class SnakeBody : GameObject
{
    public E_SneakBodyType type;

    public SnakeBody(E_SneakBodyType type, int x, int y)
    {
        this.type = type;
        pos = new Position(x, y);
    }
    public override void Draw()
    {
        Console.ForegroundColor = type == E_SneakBodyType.head ? ConsoleColor.Yellow : ConsoleColor.Green;
        Console.SetCursorPosition(pos.x, pos.y);
```

```csharp
//游戏场景
class GameScene : ISceneUpdata
{
    Map map;
    Snake snake;
    Food food;

    //降速标识
    private int updataIndex = 0;
    public GameScene()
    {
        map = new Map();
        snake = new Snake(40, 10);
        food = new Food(snake);
    }
    public void UpData()
    {
        //降速
        //用了 KeyAvailable 之后就不用设置这么大的值了
        if (updataIndex % 6000 == 0) 
        {
            map.Draw();
            //画食物 
            food.Draw();
            //蛇移动
            snake.Move();
            //画蛇
            snake.Draw();
            //检测是否撞墙
            if (snake.CheckEnd(map))
            {
	            Game.ChangeScene(E_SceneType.end);
            }

            //检测是否吃到食物
            snake.checkEatFood(food);

            //刷新
            updataIndex = 0;
        }
        ++updataIndex;

        //判断输入 有输入才会执行
        if (Console.KeyAvailable) 
        {
            //蛇转向
            switch (Console.ReadKey(true).Key)
            {
                case ConsoleKey.W:
                    snake.ChangeDir(E_MoveDir.up);
                    break;
                case ConsoleKey.A:
                    snake.ChangeDir(E_MoveDir.left);
                    break;
                case ConsoleKey.S:
                    snake.ChangeDir(E_MoveDir.down);
                    break;
                case ConsoleKey.D:
                    snake.ChangeDir(E_MoveDir.right);
                    break;
            }
        }
        
    }
}
```

### 游戏类（启动、场景切换） 

```csharp
/// <summary>
/// 场景类型枚举
/// </summary>
enum E_SceneType
{
    begin,
    game,
    end,
}
class Game
{
    //游戏窗口宽高
    //因为宽和高本身不会改变 而且会被经常调用 所以将其生成为常量 方便在其他类中直接调用
    public const int w = 80;
    public const int h = 20;
    //当前选中场景
    //修饰为静态成员
    public static ISceneUpdata nowScene;

    //控制台初始化
    public Game()
    {
        Console.CursorVisible = false;
        Console.SetWindowSize(w, h);
        Console.SetBufferSize(w, h);

        //测试场景
        //ChangeScene(E_SceneType.game);
        //设置默认场景 开始页面
        ChangeScene(E_SceneType.begin);
    }
    //游戏开始的方法
    public void Start()
    {
        //游戏主循环 主要负责 游戏场景逻辑的更新
        while (true) 
        {
            //判断当前场景不为空 就更新
            if (nowScene != null) 
            {
                nowScene.UpData();
            }
        }
    }
    //切换场景的方法
    //因为这个方法需要经常调用 所以可以将其修饰为静态方法
    public static void ChangeScene(E_SceneType type)
    {
        //切换场景前 先清空控制台
        Console.Clear();
        //修饰为静态方法后不能直接使用成员变量 所以可以将nowScene也修饰为静态成员
        switch (type)
        {
            case E_SceneType.begin:
                nowScene = new BeginScene();
                break;
            case E_SceneType.game:
                nowScene = new GameScene();
                break;
            case E_SceneType.end:
                nowScene = new EndScene();
                break;
        }
    }
}

//运行游戏
Game game = new Game();
game.Start();
```

```csharp
class Map : IDraw
{
    //暂时设为私人的 如果需要调用再改
    public Wall[] walls;
    public Map()
    {
        //初始化 将数组设定为墙的数量
        walls = new Wall[Game.w + (Game.h - 2) * 2];
        //计数
        int index = 0;
        for (int i = 0; i < Game.w; i+=2) 
        {
            walls[index] = new Wall(i, 0);
            ++index;
            walls[index] = new Wall(i, Game.h - 1);
            ++index;
        }
        for (int i = 1; i < Game.h - 1; i++) 
        {
            walls[index] = new Wall(0, i);
            ++index;
            walls[index] = new Wall(Game.w - 2, i);
            ++index;
        }
    }

    //画地图的方法
    public void Draw()
    {
        for (int i = 0; i < walls.Length; i++) 
        {
            walls[i].Draw();
        }
    }
}
```

#### 蛇类

```csharp
//蛇类
/// <summary>
/// 移动方向
/// </summary>
enum E_MoveDir
{
    up,
    down, 
    left, 
    right,
}

internal class Snake : IDraw
{
    //蛇的身子可以用数组表示
    SnakeBody[] body;
    E_MoveDir dir;
    //计数 当前身体长度
    int nowNum;
    public Snake(int x, int y)
    {
        //最粗暴的方法 把蛇身子数组容量设置成200
        body = new SnakeBody[200];
        body[0] = new SnakeBody(E_SneakBodyType.head, x, y);
        //默认移动方向
        dir = E_MoveDir.right;
        nowNum = 1;
    }
    //画身子
    public void Draw()
    {
        //数组长度已经定好 所以用nowNum计数
        for (int i = 0; i < nowNum; i++) 
        {
            body[i].Draw();
        }
    }
    //蛇移动
    public void Move()
    {
        //移动之前需要擦除 最后一个位置
        SnakeBody lastBody = body[nowNum - 1];
        Console.SetCursorPosition(lastBody.pos.x, lastBody.pos.y);
        Console.Write("  ");
        //身体的移动 新长出来的身体 等于上一个身体
        for (int i = nowNum - 1; i > 0; i--) 
        {
            body[i].pos = body[i - 1].pos;
        }
        switch (dir)
        {
            case E_MoveDir.up:
                --body[0].pos.y;
                break;
            case E_MoveDir.down:
                ++body[0].pos.y;
                break;
            case E_MoveDir.left:
                body[0].pos.x -= 2;
                break;
            case E_MoveDir.right:
                body[0].pos.x += 2;
                break;
        }
    }
    //蛇转向
    public void ChangeDir(E_MoveDir dir)
    {
        //只有头部的时候 左转右 右转左 上转下 下转上
        //否则 不能转 传入的方向和当前方向一样也不能转
        if (dir == this.dir || nowNum > 1 &&
            (this.dir == E_MoveDir.right && dir == E_MoveDir.left ||
             this.dir == E_MoveDir.left && dir == E_MoveDir.right ||
             this.dir == E_MoveDir.up && dir == E_MoveDir.down ||
             this.dir == E_MoveDir.down && dir == E_MoveDir.up))
        {
            return;
        }
        this.dir = dir;
    }
    //蛇撞墙/检测是否结束游戏
    public bool CheckEnd(Map map)
    {
        //蛇头位置与墙位置重合
        for (int i = 0; i < map.walls.Length; i++) 
        {
            if (body[0].pos == map.walls[i].pos)
            {
                return true;
            }
        }
        //蛇头位置与蛇身位置重合
        for (int i = 1; i < nowNum; i++)
        {
            if (body[0].pos == body[i].pos )
            {
                return true;
            }
        }
        return false;
    }
    //传入一个位置 判断是不是和蛇重合
    public bool checkSamePos(Position p)
    {
        for (int i = 1; i < nowNum; i++)
        {
            if (p == body[i].pos)
            {
                return true;
            }
        }
        return false;
    }
    //判断蛇头是不是与食物重合
    //吃到了 再随机位置 增加蛇的长度
    public void checkEatFood(Food food)
    {
        if (body[0].pos == food.pos)
        {
            //把自己传进去
            food.randomPos(this);

            //增加蛇的长度
            AddBody();
        }
    }
    //增加蛇的长度
    private void AddBody()
    {
        //生成蛇之前的位置
        SnakeBody frontBody = body[nowNum - 1];
        //先加一节身体 等于上一节身体
        body[nowNum] = new SnakeBody(E_SneakBodyType.body, frontBody.pos.x, frontBody.pos.y);
        //再加长度
        ++nowNum;
    }
}
```
