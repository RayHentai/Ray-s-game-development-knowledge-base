### 练习

> 1. 在上节课的练习题基础上,请用现在所学知识，制作一个这样的功能：场景上对象发射子弹有音效，通过用UGUI的Toggle开关控制音效开关

```csharp
public class Toggle_MusicData
{
    public static bool isOpen = true;
}

public class Toggle_Panel : MonoBehaviour
{
    public Toggle_Player player;
    public Button btnAtk;
    public Toggle on;
    public Toggle off;
    public ToggleGroup tg;
    void Start()
    {
        btnAtk.onClick.AddListener(() => { player.Fire(); });
        on.onValueChanged.AddListener(TogChangeValue);
        on.onValueChanged.AddListener(TogChangeValue);
    }
    private void TogChangeValue(bool value) 
    {
        foreach (Toggle item in tg.ActiveToggles())
        {
            if (item == on)
                Toggle_MusicData.isOpen = true;
            else if (item == off)
                Toggle_MusicData.isOpen = false;
        }
    }
}

public class Toggle_Player : MonoBehaviour
{
    public AudioClip clip;
    public void Fire()
    {
        if (Toggle_MusicData.isOpen)
        {
            AudioSource audioSource = this.AddComponent<AudioSource>();
            audioSource.clip = clip;
            audioSource.Play();
            Destroy(audioSource, 0.8f);
        }
        Instantiate(Resources.Load<GameObject>("Bullet"), transform.position,transform.rotation);
    }
}
```

public class Slider_MusicData
{
    public static bool isOpen = true;
    public static float musicValue = 1;
}
public class Slider_Panel : MonoBehaviour
{
    private static Slider_Panel instance;
    public static Slider_Panel Instance => instance;
    public Slider_Player player;
    public Button btnAtk;
    public Toggle on;
    public Toggle off;
    public ToggleGroup tg;
    public Button btnChangeName;
    public Text textName;
    public Slider slider;
    private void Awake()
    {
        instance = this;
    }
    void Start()
    {
        btnAtk.onClick.AddListener(() => 
        {
            player.Fire(); 
        });
        on.onValueChanged.AddListener(TogChangeValue);
        on.onValueChanged.AddListener(TogChangeValue);
        btnChangeName.onClick.AddListener(() =>
        {
            Slider_ChangeName.Instance.gameObject.SetActive(true);
        });
        slider.onValueChanged.AddListener((a)=> { Slider_MusicData.musicValue = a; });
    }
    private void TogChangeValue(bool value) 
    {
        foreach (Toggle item in tg.ActiveToggles())
        {
            if (item == on)
                Slider_MusicData.isOpen = true;
            else if (item == off)
                Slider_MusicData.isOpen = false;
        }
    }
}
public class Slider_Player : MonoBehaviour
{
    public AudioClip clip;
    public void Fire()
    {
        if (Slider_MusicData.isOpen)
        {
            AudioSource audioSource = this.AddComponent<AudioSource>();
            audioSource.clip = clip;
            audioSource.volume = Slider_MusicData.musicValue;
            audioSource.Play();
            Destroy(audioSource, 0.8f);
        }
        GameObject obj = Instantiate(Resources.Load<GameObject>("Bullet"), transform.position, transform.rotation);
        Destroy(obj, 5);

## Scroll View 滚动视图

> Scroll Rect是滚动视图组件，是UGUI中用于处理滚动视图相关交互的关键组件

**默认创建的Scroll Rect由4组对象组成**
- 父对象一ScrollRect组件依附的对象还有一个Image组件最为背景图
- 子对象
