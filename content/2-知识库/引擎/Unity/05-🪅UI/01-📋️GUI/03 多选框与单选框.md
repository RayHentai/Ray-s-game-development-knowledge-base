## 多选框 Toggle

```csharp
// 基本使用
// 参数二：选中/未选中状态，一定要在外面声明一个 bool 变量存储状态
// 返回值 bool：多选框的当前选中状态
bool isSel = false;

void OnGUI()
{
    isSel = GUI.Toggle(rect, isOn, "选项名称");
    isSel = GUI.Toggle(rect, isOn, content, style);  // 综合 + 自定义样式
}
```

## 多选框样式问题

> 直接调整宽高会影响响应区域而非显示大小，需要通过 GUIStyle 来控制。

```
在 GUIStyle 中设置：
- fixedWidth / fixedHeight  → 修改固定宽高（不影响响应区域）
- padding → 修改从 GUIStyle 边缘到内容起始处的空间
```

## 单选框（基于多选框实现）

```csharp
// 通过 int 标识决定选中哪一个（索引值相同则选中）
int selectedIndex = 0;

void OnGUI()
{
    // 将当前值和代表该选项的 int 比较：相等则为选中
    if (GUI.Toggle(rect1, selectedIndex == 0, "选项A"))
        selectedIndex = 0;
    if (GUI.Toggle(rect2, selectedIndex == 1, "选项B"))
        selectedIndex = 1;
    if (GUI.Toggle(rect3, selectedIndex == 2, "选项C"))
        selectedIndex = 2;
}
```

## 练习

> 1. 在上一知识点练习题基础上制作一个游戏设置面板，上面有背景音乐的开关和音效的开关，并且点击开始界面的"设置"按钮可以打开设置面板

```csharp
public class Button : MonoBehaviour
{
    //类似单例模式的静态变量
    private static Button instance;
    //隐藏窗口
    public static void ShowMe() 
    {
        //保护 防止报错
        if (instance != null)
        {
            instance.gameObject.SetActive(true);
        }
    }
    //显示窗口
    public static void HidMe()
    {
        if (instance != null)
            instance.gameObject.SetActive(false);
    }
    public Rect titlePos;
    public GUIContent content;
    public GUIStyle style;

    public Rect button1Pos;
    public Rect button2Pos;
    public Rect button3Pos;
    public GUIStyle buttonStyle;
    //启动之际就让静态变量等于自己
    private void Awake()
    {
        instance = this; 
        //显示窗口的同时隐藏设置窗口
        //因为暂时还没办法设置窗口优先级 只能这样处理
        Setting.HidMe();
    }
    private void OnGUI()
    {
        GUI.Label(titlePos, content, style);
        if (GUI.Button(button1Pos, "开始游戏", buttonStyle))
            SceneManager.LoadScene("Game");
        if (GUI.Button(button2Pos, "设置", buttonStyle))
        {
            //调用设置窗口的显示 再关闭自己
            Setting.ShowMe();
            HidMe();
        }
        if (GUI.Button(button3Pos, "结束游戏", buttonStyle))
        {
			
        }
    }
}

public class Setting : MonoBehaviour
{
    private static Setting instance;
    public static void ShowMe()
    {
        if (instance != null)
            instance.gameObject.SetActive(true);
    }
    public static void HidMe()
    {
        if (instance != null)
            instance.gameObject.SetActive(false);
    }
    public Rect pos1;
    public Rect pos2;
    public Rect pos3;

    bool a;
    bool b;

    public GUIStyle style;
    public GUIStyle style1;

    private void Awake()
    {
        instance = this;
    }
    private void OnGUI()
    {
        if (GUI.Button(pos3, " ", style1))
        {
            HidMe();
            Button.ShowMe();
        }
        a = GUI.Toggle(pos1, a, "背景音乐", style);
        b = GUI.Toggle(pos2, b, "背景音乐", style);

    }
}

```
