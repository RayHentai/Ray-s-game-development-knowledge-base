## 输入框 TextField / PasswordField

string inputText = "";

void OnGUI()
{
    // 普通输入框（返回输入后的字符串）
    inputText = GUI.TextField(rect, inputText);
    inputText = GUI.TextField(rect, inputText, 最大输入长度);
    inputText = GUI.TextField(rect, inputText, 最大输入长度, style);

    // 密码输入框（显示为遮罩字符）
    password = GUI.PasswordField(rect, password, '*');
    password = GUI.PasswordField(rect, password, '*', 最大输入长度, style);
}

## 拖动条 Slider

```csharp
float sliderValue = 50f;

void OnGUI()
{
    // 水平拖动条（三个值均为 float，返回 float）
    sliderValue = GUI.HorizontalSlider(rect, sliderValue, 最小值, 最大值);
    sliderValue = GUI.HorizontalSlider(rect, sliderValue, 0f, 100f, style);

    // 竖直拖动条
    sliderValue = GUI.VerticalSlider(rect, sliderValue, 最小值, 最大值);
    sliderValue = GUI.VerticalSlider(rect, sliderValue, 0f, 100f, style);
}
```

> 1. 在设置面板中加入拖动条控制音乐大小

public class Setting : MonoBehaviour
{
    public Rect VocalPos1;

        nowVocal1 = GUI.HorizontalSlider(VocalPos1, nowVocal1, 0, 100);
        nowVocal2 = GUI.HorizontalSlider(VocalPos2, nowVocal2, 0, 100);

> 2. 制作一个登录面板：有用户名输入和密码输入，有进入游戏和返回上一级按钮，点击开始界面"开始游戏"后进入登录面板，输入用户名 admin、密码 8888 才能切换场景

```csharp
public class Login : MonoBehaviour
{
    private static Login instance;
    public static void ShowMe()
    {
        if (instance != null)
        {
            instance.gameObject.SetActive(true);
        }
    }
    public static void HidMe()
    {
        if (instance != null)
            instance.gameObject.SetActive(false);
    }
    public Rect titlePos;
    public GUIContent content;
    public GUIStyle style;

    public Rect pos1;
    public Rect pos2;
    public GUIStyle loginStyle;

    public string userID = "请输入账号";
    public string password = "请输入密码";
    public Rect pos3;
    public Rect pos4;

    public Rect pos5;
    public Rect pos6;
    public GUIStyle style2;

    private void Awake()
    {
        instance = this;
        Setting.HidMe();
    }
    private void OnGUI()
    {
        GUI.Label(titlePos, content, style);
        GUI.Label(pos1, "账号", loginStyle);
        GUI.Label(pos2, "密码", loginStyle);
        userID = GUI.TextField(pos3, userID);
        password = GUI.PasswordField(pos4, password, '*');
        if (GUI.Button(pos5, "返回", style2))
        {
            Button.ShowMe();
            HidMe();
        }
        if (GUI.Button(pos6, "进入游戏", style2))
        {
            if (userID == "admin" && password == "8888")
            {
                SceneManager.LoadScene("Game");
            }
            else
            {
                Debug.Log("用户名或密码输入错误");
            }
        }
    }
}
```
