```csharp
public class BasePanel<T> : MonoBehaviour where T : class  //[[06 泛型约束]]
{
    //单例成员
    //私有的静态成员变量
    private static T instance;
    //公共的静态成员属性或方法
    public static T Instance => instance;
    private void Awake()
    {
        //因为 继承MonoBehaviour 的 类 不能实例化新对象
        //所以 这里通过 Awake生命周期函数 对 instance 进行 实例化
        //由于 实例化出来的对象 是 该基类下的 子类
        //所以用 as 转化成对应 的 子类
        instance = this as T;
    }
    public virtual void ShowMe()
    {
        gameObject.SetActive(true);
    }
    public virtual void HidMe()
    {
        gameObject.SetActive(false);
    }
}
```

```csharp
            {
```

```csharp
public abstract class TankBaseObj : MonoBehaviour
{
    public int atk;
    public int def;
    public int maxHp;
    public int nowHp;
    public float moveSpeed = 10;
    public float roundSpeed = 100;
    public float headRoundSpeed = 100;

    //死亡特效对象 关联特效预设体
    public GameObject deadEff;
    private AudioSource audioSource;
    //攻击方法
    public abstract void Fire();

    //写到一半发现还有共同特征 补充
    public Transform tankHead;

    //受伤方法
    public virtual void Wound(TankBaseObj other) 
    {
        //计算伤害
        int dem = other.atk - def;
        if (dem <= 0) 
            return;
        nowHp-= dem;
        //血量为空执行死亡逻辑
        if (nowHp <= 0)
        {
            //防止出现显示错误
            nowHp = 0;
            Dead();
        }
    }

    //死亡方法
    public virtual void Dead()
    {
        //移除对象
        Destroy(gameObject);
        //添加死亡特效
        if (deadEff != null)
        {
            Instantiate(deadEff.gameObject, transform.position, transform.rotation);
            //设置音效
            audioSource = deadEff.GetComponent<AudioSource>();
            audioSource.volume = GameDataMgr.Instance.musicData.SoundValue;
            audioSource.mute = !GameDataMgr.Instance.musicData.isOpenSound;
            //保险起见 播放音效
            audioSource.Play();
        }
    }
}
```

```csharp
public class PlayerObj : TankBaseObj
{
    //当前装备的武器
    public WeaponObj nowWeapon;

    //武器生成的位置
    public Transform weaponPos;
    
    void Update()
    {
        //移动
        transform.Translate(Input.GetAxis("Vertical") * Vector3.forward * moveSpeed * Time.deltaTime);
        //转向
        transform.Rotate(Input.GetAxis("Horizontal") * Vector3.up * roundSpeed * Time.deltaTime);
        //炮台旋转
        tankHead.transform.Rotate(Input.GetAxis("Mouse X") * Vector3.up * headRoundSpeed * Time.deltaTime);
        //开火
        if (Input.GetMouseButtonDown(0))
        {
            Fire();
        }
    }
    public override void Fire()
    {
        //命令炮管开火
        if (nowWeapon != null)
            nowWeapon.Fire();
    }
    public void ChangeWeapon(GameObject weapon)
    {
        if (nowWeapon != null)
        {
            //删除武器 置空
            Destroy(nowWeapon.gameObject);
            nowWeapon = null;
        }
        //切换武器
        GameObject weaponObj = Instantiate(weapon, weaponPos, false);
        //记录当前武器
        nowWeapon = weaponObj.GetComponent<WeaponObj>();
        //设置父亲
        nowWeapon.SetFatherObj(this);

    }
    public override void Wound(TankBaseObj other)
    {
        base.Wound(other);
        //更新面板
        GamePanel.Instance.UpdateHP(maxHp, nowHp);
    }
    public override void Dead()
    {
        Time.timeScale = 0;
        LosePanel.Instance.ShowMe();
    }
}
```

```csharp
public class MonsterObj : TankBaseObj
{
    //朝向的目标
    private Transform targetPos;
    //随机移动的点
    public Transform[] movePos;

    //炮口朝向的对象
    public Transform targetObj;
    //与玩家的距离
    public float fireDis = 10;
    //开火计时
    private float nowTime;
    //开火间隔
    public int time = 1;

    //子弹
    public GameObject bullet;
    //开火位置
    public Transform[] firePos;

    //绘制血条
    public Texture bkHp;
    public Texture Hp;
    //血条位置、背景位置
    private Rect bkPos = new Rect();
    private Rect hpPos = new Rect();
    //缩放比例
    private float scale;

    //血条显示时间
    public float hpShowTime = 0;
    void Start()
    {
        RandomPos();
    }
    void Update()
    {
        //看向目标点
        transform.LookAt(targetPos);
        //不停朝面朝向移动
        transform.Translate(Vector3.forward * moveSpeed * Time.deltaTime);
        //如果坦克到达随机点附近 重新生成随机点
        if (Vector3.Distance(transform.position, targetPos.position) < 0.05f)
        {
            RandomPos();
        }
        //间隔射击
        tankHead.LookAt(targetObj);
        if (Vector3.Distance(targetObj.position, transform.position) < fireDis)
        {
            nowTime += Time.deltaTime;
            if (nowTime >= time)
            {
                Fire();
                nowTime = 0;
            }
        }
    }
    //随机位置
    private void RandomPos()
    {
        if (movePos.Length == 0)
            return;
        targetPos = movePos[Random.Range(0, movePos.Length)];
    }

    public override void Fire()
    {
        for (int i = 0; i < firePos.Length; i++)
        {
            GameObject obj = Instantiate(bullet, firePos[i].position, firePos[i].rotation);
            BulletObj bull = obj.GetComponent<BulletObj>();
            bull.SetFatherObj(this);
        }
    }
    //重写死亡逻辑 给玩家加分
    public override void Dead()
    {
        base.Dead();
        GamePanel.Instance.AddScore(100);
    }
    private void OnGUI()
    {
        if (hpShowTime > 0)
        {
            hpShowTime -= Time.deltaTime;
            //坐标转换 世界转屏幕
            Vector3 pos = Camera.main.WorldToScreenPoint(transform.position);
            //屏幕转GUI
            pos.y = Screen.height - pos.y;
            pos.z = Vector3.Distance(targetObj.position, transform.position) / 4;
            //调整位置
            bkPos.x = pos.x - 50 / pos.z;
            bkPos.y = pos.y - 50 / pos.z;
            //设置宽高
            bkPos.width = 100 / pos.z;
            bkPos.height = 20 / pos.z;
            GUI.DrawTexture(bkPos, bkHp);
            hpPos.x = pos.x - 50 / pos.z;
            hpPos.y = pos.y - 50 / pos.z;
            hpPos.width = (float)nowHp / maxHp * 100f / pos.z;
            hpPos.height = 20 / pos.z;
            GUI.DrawTexture(hpPos, Hp);
        }
    }
    public override void Wound(TankBaseObj other)
    {
        base.Wound(other);
        hpShowTime = 3;
    }
}
```

```csharp
public class BulletObj : MonoBehaviour
{
    public float moveSpeed = 50;

    //子弹的发射者
    public TankBaseObj fatherObj;

    //特效对象
    public GameObject effObj;

    void Update()
    {
        transform.Translate(Vector3.forward * moveSpeed * Time.deltaTime);
    }
    private void OnTriggerEnter(Collider other)
    {
        //子弹碰撞逻辑
        //打上标签 碰撞只针对部分对象
        //这里的意思是 当发射者是玩家、被射者是怪物，或者颠倒过来才会进逻辑
        if (other.CompareTag("Cube") || 
            other.CompareTag("Player") && fatherObj.CompareTag("Monster") ||
            other.CompareTag("Monster") && fatherObj.CompareTag("Player"))
        {
            //判断是否受伤
            TankBaseObj obj = other.GetComponent<TankBaseObj>();
            //这个判断的意思是 利用里氏替换原则 父类装子类 如果能得到坦克的脚本 那就是被击中的对象
            if (obj != null)
            {
                obj.Wound(fatherObj);
            }
            //爆炸特效
            if (effObj != null)
            {
                GameObject eff = Instantiate(effObj, transform.position, transform.rotation);
                //音效处理
                AudioSource audioSource = eff.GetComponent<AudioSource>();
                audioSource.volume = GameDataMgr.Instance.musicData.SoundValue;
                audioSource.mute = !GameDataMgr.Instance.musicData.isOpenSound;
            }
            Destroy(gameObject);
        }
        
        
        
    }
    //设置子弹发射者脚本
    public void SetFatherObj(TankBaseObj obj)
    {
        fatherObj = obj;
    }
}
```

```csharp
public class GameDataMgr [[单例模式]]
{
    private static GameDataMgr instance = new GameDataMgr();
    public static GameDataMgr Instance => instance;

    public MusicData musicData;
    public RankList rankData;
    private GameDataMgr()
    {
        //只要new新对象 就读取数据
        musicData = PlayerPrefsDataMgr.Instance.LoadData(typeof(MusicData), "Music") as MusicData;
        //首次读取没有数据 所以先要给MusicData进行初始化
        if (!musicData.notFirst)
        {
            musicData.notFirst = true;
            musicData.isOpenBk = true;
            musicData.isOpenSound = true;
            musicData.bkValue = 1;
            musicData.SoundValue = 1;
            //保存
            PlayerPrefsDataMgr.Instance.SaveData(musicData, "Music");
        }
        rankData = PlayerPrefsDataMgr.Instance.LoadData(typeof(RankList), "Rank") as RankList;
    }
    //给外部提供设置界面改变数据的方法
    public void OpenOrCloseBKMusic(bool isOpen)
    {
        musicData.isOpenBk = isOpen;
        PlayerPrefsDataMgr.Instance.SaveData(musicData, "Music");
        //添加上声音管理器提供的方法 下面同理
        BKMusic.Instance.ChangeOpen(musicData.isOpenBk);
    }
    public void OpenOrCloseSound(bool isOpen)
    {
        musicData.isOpenSound = isOpen;
        PlayerPrefsDataMgr.Instance.SaveData(musicData, "Music");
    }
    public void ChangeBKValue(float value)
    {
        musicData.bkValue = value;
        PlayerPrefsDataMgr.Instance.SaveData(musicData, "Music");
        BKMusic.Instance.ChangeValue(musicData.bkValue);
    }
    public void ChangeSoundValue(float value)
    {
        musicData.SoundValue = value;
        PlayerPrefsDataMgr.Instance.SaveData(musicData, "Music");
    }
    //提供一个在排行榜添加数据的方法
    public void AddRanInfo(string name, int score, float time)
    {
        rankData.list.Add(new RankInfo(name, score, time));
        //根据时间排序
        rankData.list.Sort((a, b) => a.time < b.time ? -1 : 1);
        //清除10条以后的数据
        for (int i = rankData.list.Count - 1; i >= 10; i--)
        {
            rankData.list.RemoveAt(i);
        }
        //存储
        PlayerPrefsDataMgr.Instance.SaveData(rankData, "Rank");
    }
}
```
