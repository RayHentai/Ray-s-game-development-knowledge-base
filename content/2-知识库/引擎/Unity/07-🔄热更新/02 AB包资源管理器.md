```csharp
/// <summary>
/// AB包管理器
/// </summary>
public class ABMgr : SingletonAutoMono<ABMgr>
{
    private ABMgr() { }
    private Dictionary<string, AssetBundle> abDic = new Dictionary<string, AssetBundle>(); //容器 用来存储加载过的包
    private AssetBundle mainAB = null; //主包
    private AssetBundleManifest manifest = null; //主包的配置文件
    private string PathUrl => Application.streamingAssetsPath + "/"; //主包路径
    private string MainABName //不同平台的主包路径
    {
        get 
        {
            #if UNITY_IOS
            return "IOS"
            #elif UNITY_ANDROID
            return "Android"
            #else
            return "PC";
            #endif
        }
    }
    /// <summary>
    /// 加载AB包
    /// </summary>
    /// <param name="abName">AB包名</param>
    public void LoadAB(string abName) 
    {
        AssetBundle ab = null;
        if (mainAB == null) //如果主包为空
        {
            mainAB = AssetBundle.LoadFromFile(PathUrl + MainABName); //加载主包
            manifest = mainAB.LoadAsset<AssetBundleManifest>("AssetBundleManifest"); //加载主包的配置文件
        }
        string[] strs = manifest.GetAllDependencies(abName);//得到依赖包信息
        for (int i = 0; i < strs.Length; i++) 
        {
            if (!abDic.ContainsKey(strs[i])) //如果字典中没有该包
            {
                ab = AssetBundle.LoadFromFile(PathUrl + strs[i]); //加载依赖包
                abDic.Add(strs[i], ab); //放入字典
            }
        }
        if (!abDic.ContainsKey(abName)) //如果字典不为空 
        {
            ab = AssetBundle.LoadFromFile(PathUrl + abName);
            abDic.Add(abName, ab);
        }
    }
    /// <summary>
    /// 同步加载AB包资源（字符串重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <returns></returns>
    public UnityEngine.Object LoadRes(string abName, string resName) 
    {
        LoadAB(abName); //加载AB包
        UnityEngine.Object obj = abDic[abName].LoadAsset(resName); //得到该资源对象
        if (obj is GameObject) //如果该对象是GameObject
            return Instantiate(obj); //实例化
        else //否则
            return obj; //直接返回
    }
    /// <summary>
    /// 同步加载AB包资源（Type重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="type">资源类型</param>
    /// <returns></returns>
    public UnityEngine.Object LoadRes(string abName, string resName, Type type)
    {
        LoadAB(abName); //加载AB包
        UnityEngine.Object obj = abDic[abName].LoadAsset(resName, type) ; //得到该资源对象
        if (obj is GameObject) //如果该对象是GameObject
            return Instantiate(obj); //实例化
        else //否则
            return obj; //直接返回
    }
    /// <summary>
    /// 同步加载AB包资源（泛型重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <returns></returns>
    public T LoadRes<T>(string abName, string resName) where T : UnityEngine.Object
    {
        LoadAB(abName); //加载AB包
        T obj = abDic[abName].LoadAsset<T>(resName); //得到该资源对象
        if (obj is GameObject) //如果该对象是GameObject
            return Instantiate(obj); //实例化
        else //否则
            return obj; //直接返回
    }
    /// <summary>
    /// 异步加载AB包资源（字符串重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="callback">回调函数</param>
    public void LoadResAsync(string abName,string resName, UnityAction<UnityEngine.Object> callback)
    {
        StartCoroutine(ReallyLoadResAsync(abName, resName, callback));
    }
    /// <summary>
    /// 真正异步加载AB包资源（字符串重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="callback">回调函数</param>
    /// <returns></returns>
    private IEnumerator ReallyLoadResAsync(string abName, string resName, UnityAction<UnityEngine.Object> callback) 
    {
        LoadAB(abName); //加载AB包
        AssetBundleRequest abr = abDic[abName].LoadAssetAsync(resName);//异步加载
        yield return abr;
        //加载结束
        if (abr.asset is GameObject) //如果对象是GameObject 实例化并传入回调函数
            callback(Instantiate(abr.asset));
        else
            callback(abr.asset);//直接传入回调函数
    }
    /// <summary>
    /// 异步加载AB包资源（Type重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="type">资源类型</param>
    /// <param name="callback">回调函数</param>
    public void LoadResAsync(string abName, string resName, Type type,UnityAction<UnityEngine.Object> callback)
    {
        StartCoroutine(ReallyLoadResAsync(abName, resName, type, callback));
    }
    /// <summary>
    /// 真正异步加载AB包资源（Type重载）
    /// </summary>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="type">资源类型</param>
    /// <param name="callback">回调函数</param>
    private IEnumerator ReallyLoadResAsync(string abName, string resName, Type type,UnityAction<UnityEngine.Object> callback)
    {
        LoadAB(abName); //加载AB包
        AssetBundleRequest abr = abDic[abName].LoadAssetAsync(resName);//异步加载
        yield return abr;
        //加载结束
        if (abr.asset is GameObject) //如果对象是GameObject 实例化并传入回调函数
            callback(Instantiate(abr.asset));
        else
            callback(abr.asset);//直接传入回调函数
    }
    /// <summary>
    /// 异步加载AB包资源（泛型重载）
    /// </summary>
    /// <typeparam name="T">资源</typeparam>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="callback">回调函数</param>
    public void LoadResAsync<T>(string abName, string resName, UnityAction<T> callback) where T : UnityEngine.Object
    {
        StartCoroutine(ReallyLoadResAsync<T>(abName, resName, callback));
    }
    /// <summary>
    /// 真正异步加载AB包资源（泛型重载）
    /// </summary>
    /// <typeparam name="T">资源</typeparam>
    /// <param name="abName">AB包名</param>
    /// <param name="resName">资源名</param>
    /// <param name="callback">回调函数</param>
    private IEnumerator ReallyLoadResAsync<T>(string abName, string resName, UnityAction<T> callback) where T : UnityEngine.Object
    {
        LoadAB(abName); //加载AB包
        AssetBundleRequest abr = abDic[abName].LoadAssetAsync<T>(resName);//异步加载
        yield return abr;
        //加载结束
        if (abr.asset is GameObject) //如果对象是GameObject 实例化并传入回调函数
            callback(Instantiate(abr.asset) as T);
        else
            callback(abr.asset as T);//直接传入回调函数
    }
    /// <summary>
    /// 卸载单个包
    /// </summary>
    /// <param name="abName">包名</param>
    public void UnLoad(string abName)
    {
	    if (abDic.ContainsKey(abName)) //如果字典中存在包 
	    {
		    abDic[abName].Unload(false);//卸载包
	        abDic.Remove(abName);//移除字典中的记录
	    }
        
    }
    /// <summary>
    /// 卸载所有包
    /// </summary>
    public void ClearAB()
    {
        AssetBundle.UnloadAllAssetBundles(false);//卸载所有包
        abDic.Clear(); //清空容器
        mainAB = null; //清空主包
        manifest = null; //清空主包配置文件
    }
}
```
