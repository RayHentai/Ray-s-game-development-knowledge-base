```csharp
public abstract class BasePanel<T> : MonoBehaviour where T : class
{
    private static T Instance;
    public static T instance => Instance;
    private void Awake()
    {
        Instance = this as T;
    }
    void Start()
    {
        //子类不用主动调用这个函数 只要实现了Init方法 就会自动在这里执行
        Init();
    }
    //这个函数主要用于显示的初始化
    public abstract void Init();
    public virtual void ShowMe()
    {
        gameObject.SetActive(true);
    }
    public virtual void HideMe() 
    {
        gameObject.SetActive(false);
    }
}
```

```csharp
public class ChoosePanel : BasePanel<ChoosePanel>
{
    public UIButton btnLeft;
    public UIButton btnRight;
    public UIButton btnClose;
    public UIButton btnStart;
    //模型的父对象
    public Transform fatherObj;
    //模型
    public GameObject planeObj;
    //模型的属性信息代表的控件
    public List<GameObject> hpObjs;
    public List<GameObject> speedObjs;
    public List<GameObject> volumeObjs;
    private float time;
    private bool isSel = false;

    public override void Init()
    {
        btnLeft.onClick.Add(new EventDelegate(() => 
        { 
            --GameDataMgr.Instance.nowSelPlane;
            if (GameDataMgr.Instance.nowSelPlane < 0)
                GameDataMgr.Instance.nowSelPlane = GameDataMgr.Instance.roleData.RoleList.Count - 1;
            ChangePlane(); 
        }));
        btnRight.onClick.Add(new EventDelegate(() => 
        {
            ++GameDataMgr.Instance.nowSelPlane;
            if (GameDataMgr.Instance.nowSelPlane > GameDataMgr.Instance.roleData.RoleList.Count - 1)
                GameDataMgr.Instance.nowSelPlane = 0;
            ChangePlane(); 
        }));
        btnClose.onClick.Add(new EventDelegate(() => { HideMe(); BeginPanel.instance.ShowMe(); }));
        btnStart.onClick.Add(new EventDelegate(() => { SceneManager.LoadScene("GameScene"); }));
        HideMe();
    }
    void Update()
    {
        time += Time.deltaTime;
        //上下抖动
        planeObj.transform.Translate(Vector3.up * Mathf.Sin(time ) * 0.0001f, Space.World);
        //拖动旋转
        if (Input.GetMouseButtonDown(0))
        {
            if (Physics.Raycast(Camera.main.ScreenPointToRay(Input.mousePosition), 1000, 1<<LayerMask.NameToLayer("UI")))
                isSel = true;
        }
        if (Input.GetMouseButtonUp(0))
        {
            isSel = false;
        }
        if (Input.GetMouseButton(0) || isSel)
        {
            fatherObj.rotation = Quaternion.AngleAxis(Input.GetAxis("Mouse X"), Vector3.up) * fatherObj.rotation;
        }
    }
    public override void HideMe()
    {
        //删除模型
        DestroyObj();
        base.HideMe();
    }
    public override void ShowMe()
    {
        //置0当前选择 每次进入从第0个开始
        GameDataMgr.Instance.nowSelPlane = 0;
        ChangePlane();
        base.ShowMe();
    }
    private void ChangePlane() 
    {
        RoleInfo info = GameDataMgr.Instance.GetNowSelPlane();
        //更新模型
        DestroyObj();//先删除
        planeObj = Instantiate(Resources.Load<GameObject>(info.resName));
        planeObj.transform.SetParent(fatherObj, false);
        planeObj.transform.localPosition = Vector3.zero;
        planeObj.transform.localRotation = Quaternion.identity;
        planeObj.transform.localScale = Vector3.one * info.scale;
        planeObj.layer = LayerMask.NameToLayer("UI");
        //更新属性
        for (int i = 0; i < 10; i++) 
        {
            hpObjs[i].SetActive(i < info.hp);
            speedObjs[i].SetActive(i < info.speed);
            volumeObjs[i].SetActive(i < info.volume);
        }
    }
    private void DestroyObj()
    {
        if (planeObj != null)
        {
            Destroy(planeObj);
            planeObj = null;
        }
    }
}
```

```csharp
public class RankPanel : BasePanel<RankPanel>
{
    public UIButton btnClose;
    public UIScrollView svList;
    public List<RankItem> rankItems = new List<RankItem>();
    public override void Init()
    {
        btnClose.onClick.Add(new EventDelegate(() => { HideMe(); }));
        HideMe();
    }
    public override void ShowMe()
    {
        //更新面板
        //获取信息
        List<RankInfo> rankInfos = GameDataMgr.Instance.rankData.rankList;
        for (int i = 0; i < rankInfos.Count; i++)
        {
            //如果面板上已经存在控件 直接更新
            if (i < rankItems.Count)
            {
                rankItems[i].InitInfo(i + 1, rankInfos[i].name, rankInfos[i].time);
            }
            //否则 实例化一个新的
            else
            {
                GameObject sv = Instantiate(Resources.Load<GameObject>("UI/RankItem"));
                //设置父对象
                sv.transform.SetParent(svList.transform, false);
                //设置位置
                sv.transform.localPosition = new Vector3(0, 165 - 55 * i, 0);
                //设置数据
                RankItem info = sv.GetComponent<RankItem>();//获取脚本
                info.InitInfo(i + 1, rankInfos[i].name, rankInfos[i].time);//设置信息
                rankItems.Add(info);//添加
            }
        }
        base.ShowMe();
    }
}
```

```csharp
public class GameDataMgr
{
    private static GameDataMgr instance = new GameDataMgr();
    public static GameDataMgr Instance => instance;
    public MusicData musicData;
    public RankData rankData = new RankData();
    public RoleData roleData = new RoleData();
    public BulletData bulletData = new BulletData();
    public FireData fireData = new FireData();
    public int nowSelPlane = 0;
    private GameDataMgr()
    {
        musicData = XmlDataMgr.Instance.LoadData(typeof(MusicData), "MusicData") as MusicData;
        rankData = XmlDataMgr.Instance.LoadData(typeof(RankData), "RankData") as RankData;
        roleData = XmlDataMgr.Instance.LoadData(typeof(RoleData), "RoleData") as RoleData;
        bulletData = XmlDataMgr.Instance.LoadData(typeof(BulletData), "BulletData") as BulletData;
        fireData = XmlDataMgr.Instance.LoadData(typeof(FireData), "FireData") as FireData;
    }

    public void SaveMusicData() { XmlDataMgr.Instance.SaveData(musicData, "MusicData"); }
    public void AddRankInfo(string name, int time)
    {
        //添加数据
        RankInfo info = new RankInfo();
        info.name = name;
        info.time = time;
        rankData.rankList.Add(info);
        //排序
        rankData.rankList.Sort((a, b) => a.time > b.time ? -1 : 1);
        //移除
        if (rankData.rankList.Count > 20)
            rankData.rankList.RemoveAt(20);
        //rankData.rankData.RemoveRange(20, rankData.rankData.Count - 20); 方法2
        XmlDataMgr.Instance.SaveData(rankData, "RankData");
    }
    public void SetMusicValue(float value)
    {
        //改数据
        musicData.musicValue = value;
        //真正控制音量大小
        BkMusic.Instance.SetValue(value);
    }
    public void SetSoundValue(float value)
    {
        musicData.soundValue = value;
    }
    public void SetIsMusicOpen(bool isOpen)
    {
        musicData.isMusicOpen = isOpen;
        BkMusic.Instance.SetIsOpen(isOpen);
    }
    public void SetIsSoundOpen(bool isOpen)
    {
        musicData.isSoundOpen = isOpen;
    }
    public RoleInfo GetNowSelPlane()
    {
        return roleData.RoleList[nowSelPlane];
    }
}
```

```csharp
public class BulletObject : MonoBehaviour

    private BulletInfo info; //子弹信息
    private float time;
    void Update()
    {
        //共同特点 面朝向移动
        transform.Translate(Vector3.forward * info.forwardSpeed * Time.deltaTime);
        //其他移动方式
        switch (info.type)
        {
            case 2://摆动
                time += Time.deltaTime;
                transform.Translate(Vector3.right * Mathf.Sin(time * info.roundSpeed ) * info.rightSpeed);
                break;
            case 3://右抛物线
                transform.rotation *= Quaternion.AngleAxis(info.roundSpeed * Time.deltaTime, Vector3.up);
                break;
            case 4://左抛物线
                transform.rotation *= Quaternion.AngleAxis(-info.roundSpeed * Time.deltaTime, Vector3.up);
                break;
            case 5://跟随移动
                transform.rotation = Quaternion.Slerp(transform.rotation, Quaternion.LookRotation(PlayerObject.Instance.transform.position - transform.position), info.roundSpeed * Time.deltaTime);
                break;
        }
    }
    private void OnTriggerEnter(Collider other)//检测碰撞
    {
        if (other.gameObject.CompareTag("Player"))
        {
            PlayerObject obj = other.GetComponent<PlayerObject>();
            obj.Wound();//玩家受伤
            Dead();//子弹死亡
        }
    }
    public void InitInfo(BulletInfo info)//初始化子弹信息
    {
        this.info = info;
        Invoke("LateDestroy", info.lifeTime);//延迟函数在对象被销毁的时候不会执行
    }
    private void LateDestroy() { Dead(); }//用于延迟移除的函数
    public void Dead()//子弹死亡
    {
        GameObject effObj = Instantiate(Resources.Load<GameObject>(info.deadEffRes));
        effObj.transform.position = transform.position;
        Destroy(effObj, 1f);//移除特效

    }
```

```csharp
public enum E_FirePos
{
    TopLeft,
    Top,
    TopRight, 
    Left,
    Right,
    BottomLeft, 
    Bottom,
    BottomRight
}
public class FireObject : MonoBehaviour
{
    public E_FirePos firePos;
    private FireInfo fireInfo;
    private BulletInfo nowBulletInfo;
    private int nowNum;
    private float nowCd;
    private float nowDelay;
    private float changeAngle;//霰弹每次改变的角度
    private Vector3 nowDir;//霰弹上一次的方向

    private Vector3 screenPos;
    private Vector3 initDir;//初始方向向量 发射霰弹时的起始方向

    void Update()
    {
        UpDatePos();
        ResetFireInfo();
        UpDateFire();
    }
    private void UpDatePos()
    {
        screenPos.z = 2382;
        switch (firePos)
        {
            case E_FirePos.TopLeft:
                screenPos.x = 0;
                screenPos.y = Screen.height;
                initDir = Vector3.right;
                break;
            case E_FirePos.Top:
                screenPos.x = Screen.width / 2;
                screenPos.y = Screen.height;
                initDir = Vector3.right;
                break;
            case E_FirePos.TopRight:
                screenPos.x = Screen.width;
                screenPos.y = Screen.height;
                initDir = Vector3.left;
                break;
            case E_FirePos.Left:
                screenPos.x = 0;
                screenPos.y = Screen.height / 2;
                initDir = Vector3.forward;
                break;
            case E_FirePos.Right:
                screenPos.x = Screen.width;
                screenPos.y = Screen.height / 2;
                initDir = Vector3.forward;
                break;
            case E_FirePos.BottomLeft:
                screenPos.x = 0;
                screenPos.y = 0;
                initDir = Vector3.right;
                break;
            case E_FirePos.Bottom:
                screenPos.x = Screen.width / 2;
                screenPos.y = 0;
                initDir = Vector3.left;
                break;
            case E_FirePos.BottomRight:
                screenPos.x = Screen.width;
                screenPos.y = 0;
                initDir = Vector3.left;
                break;
        }
        transform.position = Camera.main.ScreenToWorldPoint(screenPos);
    }
    private void ResetFireInfo()//重置炮台数据
    {
        if (nowCd != 0 && nowNum != 0)//只有cd和数量都为0 才需要重新获取发射点数据
            return;
        if (fireInfo != null)//第一次进的时候时空的 下一次进循环就开始减少组间间隔时间
        {
            nowDelay -= Time.deltaTime;
            if (nowDelay > 0)//如果还在组间休息
                return;
        } 
        List<FireInfo> list = GameDataMgr.Instance.fireData.fireData;//得到
        fireInfo = list[Random.Range(0,list.Count)];//随机子弹
        nowNum = fireInfo.num;
        nowCd = fireInfo.cd;
        nowDelay = fireInfo.delay;//分别取出需要动态改变的内容
        string[] strs = fireInfo.ids.Split(',');
        int begin = int.Parse(strs[0]);
        int end = int.Parse(strs[1]);//得到开始和结束ID
        int RandomNum = Random.Range(begin, end + 1);
        nowBulletInfo = GameDataMgr.Instance.bulletData.bulletData[RandomNum - 1];//取出子弹信息
        if (fireInfo.type == 2)//霰弹 计算间隔角度
        {
            switch (firePos)
            {
                case E_FirePos.TopLeft:
                case E_FirePos.TopRight:
                case E_FirePos.BottomLeft:
                case E_FirePos.BottomRight:
                    changeAngle = 90f / (nowNum + 1);
                    break;
                case E_FirePos.Top:
                case E_FirePos.Left:
                case E_FirePos.Right:
                case E_FirePos.Bottom:
                    changeAngle = 180f / (nowNum + 1);
                    break;
            }
        }
    }
    private void UpDateFire()
    {
        if (nowNum == 0 && nowCd == 0)//当前处于组件间隔
            return;
        nowCd -= Time.deltaTime;
        if (nowCd > 0)//当前存在Cd
            return;
        GameObject bullet;
        BulletObject bulletObject;
        switch (fireInfo.type)
        {
            case 1:
                bullet = Instantiate(Resources.Load<GameObject>(nowBulletInfo.resName));//实例化
                bulletObject = bullet.AddComponent<BulletObject>();//动态添加脚本
                bulletObject.InitInfo(nowBulletInfo);//传入当前子弹数据
                bullet.transform.position = transform.position;//位置
                bullet.transform.rotation = Quaternion.LookRotation(PlayerObject.Instance.transform.position - transform.position);
                --nowNum;//发射一颗子弹 计数减1
                nowCd = nowNum==0 ? 0 : fireInfo.cd; //重置CD
                break;
            case 2:
                if (nowCd == 0)//无CD
                {
                    for (int i = 0; i < nowNum; i++) 
                    {
                        bullet = Instantiate(Resources.Load<GameObject>(nowBulletInfo.resName));//实例化
                        bulletObject = bullet.AddComponent<BulletObject>();//动态添加脚本
                        bulletObject.InitInfo(nowBulletInfo);//传入当前子弹数据
                        bullet.transform.position = transform.position;
                        nowDir = Quaternion.AngleAxis(changeAngle * i, Vector3.up) * initDir;
                        bullet.transform.rotation = Quaternion.LookRotation(nowDir);
                    }
                    nowCd = nowNum = 0;//一次性发射完 重置数据
                }
                else 
                {
                    for (int i = 0; i < nowNum; i++)
                    {
                        bullet = Instantiate(Resources.Load<GameObject>(nowBulletInfo.resName));//实例化
                        bulletObject = bullet.AddComponent<BulletObject>();//动态添加脚本
                        bulletObject.InitInfo(nowBulletInfo);//传入当前子弹数据
                        bullet.transform.position = transform.position;
                        nowDir = Quaternion.AngleAxis(changeAngle * (fireInfo.num - nowNum) , Vector3.up) * initDir;
                        bullet.transform.rotation = Quaternion.LookRotation(nowDir);
                        --nowNum;//发射一颗子弹 计数减1
                        nowCd = nowNum == 0 ? 0 : fireInfo.cd; //重置CD
                    }
                }
                    break;
        }
    }//发射子弹
}
```

```csharp
public class PlayerObject : MonoBehaviour
{
    private static PlayerObject instance;
    public static PlayerObject Instance => instance;
    public int nowHp;
    public int maxHp;
    public int speed = 100;
    public int roundSpeed = 20;
    public bool isDead;

    private Quaternion targetQ;//旋转目标四元数

    private float Hvalue;
    private float Vvalue;
    
    private Vector3 nowPos;
    private Vector3 frontPos;

    private void Awake()
    {
        instance = this;
    }
    void Update()
    {
        if (isDead)
            return;
        //旋转
        Hvalue = Input.GetAxisRaw("Horizontal");
        Vvalue = Input.GetAxisRaw("Vertical");
        if (Hvalue == 0)
            targetQ = Quaternion.identity;
        else
            targetQ = Hvalue > 0 ? Quaternion.AngleAxis(-20, Vector3.forward) : Quaternion.AngleAxis(20, Vector3.forward);
        transform.rotation = Quaternion.Slerp(transform.rotation, targetQ, Time.deltaTime * roundSpeed);
        //移动
        frontPos = transform.position;
        transform.Translate(Vector3.right * speed * Hvalue * Time.deltaTime,Space.World);
        transform.Translate(Vector3.forward * speed * Vvalue * Time.deltaTime, Space.World);
        //边界判断
        nowPos = Camera.main.WorldToScreenPoint(transform.position);
        if (nowPos.x <= 0 || nowPos.x >= Screen.width)
           transform.position = new Vector3(frontPos.x,transform.position.y,transform.position.z);
        if (nowPos.y <= 0 || nowPos.y >= Screen.height)
            transform.position = new Vector3(transform.position.x, transform.position.y, frontPos.z);
        if (Input.GetMouseButtonDown(0))//点击移除子弹
        {
            print("开始射线检测");
            RaycastHit hitInfo;
            if (Physics.Raycast(Camera.main.ScreenPointToRay(Input.mousePosition), out hitInfo, 10000, 1 << LayerMask.NameToLayer("Bullet")))
            {
                BulletObject bulletObject = hitInfo.transform.GetComponent<BulletObject>();
                print("开始射线检测,检测到子弹" + bulletObject.name);
                bulletObject.Dead();
            }
        }
    }
    public void Dead()
    {
        isDead = true;
        GameOverPanel.instance.ShowMe();
    }
    public void Wound()
    {
        if (isDead)
            return;
        --nowHp;
        //更新面板
        GamePanel.instance.ChangeHp(nowHp);
        if (nowHp <= 0)
            Dead();
    }
}
```
